/// Force components redraw
Void RequestComponentsRedraw() {
	declare Integer CMGame_MenuComponents_RedrawRequested for System = 0;
	CMGame_MenuComponents_RedrawRequested += 1;
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Get the value of the component redraw counter
Integer GetComponentsRedrawCounter() {
	declare Integer CMGame_MenuComponents_RedrawRequested for System = 0;
	return CMGame_MenuComponents_RedrawRequested;
}
