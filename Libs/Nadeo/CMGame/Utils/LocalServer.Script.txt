/// Check if the server is a local server
Boolean IsLocalServer() {
	declare Boolean CMGame_LocalServer_IsLocalServer for System = False;
	return CMGame_LocalServer_IsLocalServer;
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Set the the server is local or not
Void SetIsLocalServer(Boolean _IsLocalServer) {
	declare Boolean CMGame_LocalServer_IsLocalServer for System = False;
	CMGame_LocalServer_IsLocalServer = _IsLocalServer;
}
