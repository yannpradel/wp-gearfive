	// Used some GT fonts but they're not necessarily meant to stay like this, pretty arbitrarily chosen
	switch(_Font)
	{
		case "Regular" 		: return "GameFontRegular";
		case "SemiBold" 		: return "GameFontSemiBold";
		case "ExtraBold" 		: return "GameFontExtraBold";


		case "Main" 		: return "GameFontSemiBold";
		case "MainSmall" 	: return "GameFontRegular";
		case "MainSmall2"	: return "GameFontRegular";

		case "NumberSmall" 	: return "OswaldMono";
		case "NumberBig" 	: return "OswaldMono";
		case "SubTitle"	: return "GameFontSemiBold";
	}

	return "";
