/**
 *	Share lib to share information between the menu and the gamepad editor (e.g. the path of the last track created)
 */
#Const Version		"1.0.0"
#Const ScriptName	"Libs/Nadeo/TMEditor/GamepadTrackEditor/Share.Script.txt"

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// Constants
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
#Const P "LibShare_" //< Prefix used to differentiate functions/variables in the script

#Struct K_MapInfo {
	Text Path;
	Text Uid;
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
Void SetLastMapCreated(Text _MapPath, Text _MapUid) {
	declare K_MapInfo LibShare_LastMapCreated for System;
	LibShare_LastMapCreated = K_MapInfo {
		Path = _MapPath,
		Uid = _MapUid
	};
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
Void ResetLastMapCreated() {
	SetLastMapCreated("", "");
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
K_MapInfo GetLastMapCreated() {
	declare K_MapInfo LibShare_LastMapCreated for System;
	return LibShare_LastMapCreated;
}
