/// Check if the navigation in the club profile overlay should be disabled
Boolean IsClubAndProfileOverlaysNavEnabled(Text _Path) {
	return !(
		_Path == C_Path_TOTDChannelDisplay ||
		_Path == C_Path_EventDetails ||
		_Path == C_Path_Trainingdisplay ||
		_Path == C_Path_CampaignDisplay ||
		_Path == C_Path_RoomCampaignDisplay ||
		_Path == C_Path_MonthlyCampaignDisplay ||
		_Path == C_Path_ArcadeChannel ||
		_Path == C_Path_PlayMap ||
		_Path == C_Path_AgainstReplay ||
		_Path == C_Path_RoomMapListDisplay ||
		_Path == C_Path_ClubItemsActivity ||
		_Path == C_Path_MapReviewActivity ||
		_Path == C_Path_BucketItemUploads
	);
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Get the list of paths where the connection error pop-up is disabled
Text[] GetConnectionErrorPageDisabled() {
	return [
		C_Path_Null, //< Disable at launch when no routes are pushed yet
		C_Path_ConnectionError,
		C_Path_PressStart,
		C_Path_Initialization
	];
}
