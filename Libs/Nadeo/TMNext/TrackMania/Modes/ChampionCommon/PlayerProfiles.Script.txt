declare PlayerProfiles_K_PlayerProfile[Text] G_OverridePlayerProfiles;

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Get the player profiles
PlayerProfiles_K_PlayerProfile[Text] GetPlayerProfiles() {
	if (G_OverridePlayerProfiles.count > 0) return G_OverridePlayerProfiles;
	return C_Private_DefaultProfiles;
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Check if a profile for the given account id exists
Boolean PlayerProfileExists(Text _AccountId) {
	if (G_OverridePlayerProfiles.count > 0) return G_OverridePlayerProfiles.existskey(_AccountId);
	return C_Private_DefaultProfiles.existskey(_AccountId);
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Get the player profile for a given account id
PlayerProfiles_K_PlayerProfile GetPlayerProfile(Text _AccountId) {
	if (G_OverridePlayerProfiles.count > 0) return G_OverridePlayerProfiles.get(_AccountId, PlayerProfiles_K_PlayerProfile {});
	return C_Private_DefaultProfiles.get(_AccountId, PlayerProfiles_K_PlayerProfile {});
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Override the default profiles by custom ones
Void OverridePlayerProfiles(CTeam _Team0, Text _PlayerProfiles) {
	declare netwrite Integer Net_ChampionCommon_OverridePlayerProfilesUpdate for _Team0;
	declare netwrite Text Net_ChampionCommon_OverridePlayerProfiles for _Team0;
	Net_ChampionCommon_OverridePlayerProfiles = _PlayerProfiles;
	Net_ChampionCommon_OverridePlayerProfilesUpdate += 1;
	
	if (_PlayerProfiles == "") {
		G_OverridePlayerProfiles = [];
	} else {
		G_OverridePlayerProfiles.fromjson(_PlayerProfiles);
	}
}
Void ResetPlayerProfiles(CTeam _Team0, Text _PlayerProfiles) {
	OverridePlayerProfiles(_Team0, "");
}
