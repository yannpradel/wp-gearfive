/**
 *	Define difficulty levels
 */
#Const Version "1.2.0"
#Const ScriptName "Libs/Nadeo/TMGame/Utils/Difficulty.Script.txt"

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// Libraries
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
#Include "ColorLib" as CL
#Include "MathLib" as ML

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// Constants
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
#Const C_Difficulty_White 0
#Const C_Difficulty_Green 1
#Const C_Difficulty_Blue 2
#Const C_Difficulty_Red 3
#Const C_Difficulty_Black 4

#Const C_Colors [
	0 => <1., 1., 1.>,
	1 => <0.43, 0.98, 0.63>,
	2 => <0., 0.2, 1.>,
	3 => <1., 0., 0.>,
	4 => <0., 0., 0.>
]

#Const C_RoyalColors [
	0 => <1., 1., 1.>,
	1 => <0.43, 0.98, 0.63>,
	2 => <0., 0.44, 0.74>,
	3 => <1., 0., 0.>,
	4 => <0., 0., 0.>
]

#Const C_Names [
	0 => _("|Difficulty|white"),
	1 => _("|Difficulty|green"),
	2 => _("|Difficulty|blue"),
	3 => _("|Difficulty|red"),
	4 => _("|Difficulty|black")
]

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// Functions
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Get the RGB color of a royal segment
Vec3 GetRoyalSegmentColorRGB(Integer _SegmentNb) {
	return C_RoyalColors[ML::Clamp(_SegmentNb - 1, 0, C_RoyalColors.count - 1)];
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Get the Hex6 color of a royal segment
Text GetRoyalSegmentColorHex6(Integer _SegmentNb) {
	return CL::RgbToHex6(GetRoyalSegmentColorRGB(_SegmentNb));
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Get the Hex3 color of a royal segment
Text GetRoyalSegmentColorHex3(Integer _SegmentNb) {
	return CL::RgbToHex3(GetRoyalSegmentColorRGB(_SegmentNb));
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Get the name of a royal segment
Text GetRoyalSegmentName(Integer _SegmentNb) {
	return C_Names[ML::Clamp(_SegmentNb - 1, 0, C_Names.count - 1)];
}
