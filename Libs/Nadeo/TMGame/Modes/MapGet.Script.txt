/** Get the waypoint times of the author
 *	validation run
 *
 *	@return														The waypoint times
 */
Integer[] GetAuthorRaceWaypointTimes() {
	declare metadata Integer[] Race_AuthorRaceWaypointTimes for Map;
	return Race_AuthorRaceWaypointTimes;
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/** Share waypoint times of the author
 *	validation run with the map type
 *
 *	@param	_RaceWaypointTimes				The waypoint times
 */
Void ShareAuthorRaceWaypointTimes(Integer[] _RaceWaypointTimes) {
	declare Integer[] Race_Share_AuthorRaceWaypointTimes for Map;
	Race_Share_AuthorRaceWaypointTimes = _RaceWaypointTimes;
}
Integer[] GetSharedAuthorRaceWaypointTimes() {
	declare Integer[] Race_Share_AuthorRaceWaypointTimes for Map;
	return Race_Share_AuthorRaceWaypointTimes;
}
Void ShareAuthorTime(Integer _Time) {
	declare Integer Race_Share_AuthorTime for Map = -1;
	Race_Share_AuthorTime = _Time;
}
Integer GetSharedAuthorTime() {
	declare Integer Race_Share_AuthorTime for Map = -1;
	return Race_Share_AuthorTime;
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/** Share ghost id of the author
 *
 *	@param	_AuthorGhostId				The Ghost Id of the author map validation
 */
Void ShareAuthorGhostId(Ident _AuthorGhostId) {
	declare Ident Race_Share_AuthorGhostId for Map;
	Race_Share_AuthorGhostId = _AuthorGhostId;
}
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/** Get ghost id of the author
 *
 *	@return												The Ghost Id of the author map validation
 */
Ident GetAuthorGhostId() {
	declare Ident Race_Share_AuthorGhostId for Map;
	return Race_Share_AuthorGhostId;
}
