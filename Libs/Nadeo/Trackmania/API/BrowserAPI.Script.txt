Boolean FilterMap(K_Settings _Settings, Text _AuthorLogin, Boolean _IsPlayable, Text _CollectionName, Text _MapType) {
	return (
		(!_Settings.ShowOnlyUserMaps || _AuthorLogin == LocalUser.Login) &&
		(!_Settings.ShowOnlyPlayableMaps || _IsPlayable) &&
		(_Settings.FilterByEnvironment.count <= 0 || _Settings.FilterByEnvironment.exists(_CollectionName)) &&
		(_Settings.FilterByMapType.count <= 0 || _Settings.FilterByMapType.exists(_MapType))
	);
}

Boolean MapsShouldBeFiltered(K_Settings _Settings) {
	return (
		_Settings.ShowOnlyUserMaps ||
		_Settings.ShowOnlyPlayableMaps ||
		_Settings.FilterByEnvironment.count > 0 ||
		_Settings.FilterByMapType.count > 0
	);
}

CampaignStruct::LibCampaignStruct_K_Campaign FilterCampaignMaps(K_Settings _Settings, CampaignStruct::LibCampaignStruct_K_Campaign _Campaign) {
	if (!MapsShouldBeFiltered(_Settings)) return _Campaign;

	declare CampaignStruct::LibCampaignStruct_K_Campaign Campaign = _Campaign;
	Campaign.Categories = [];

	foreach (Category in _Campaign.Categories) {
		declare CampaignStruct::LibCampaignStruct_K_Category NewCategory = Category;
		NewCategory.Playlist = [];

		foreach (Map in Category.Playlist) {
			if (FilterMap(_Settings, Map.AuthorLogin, Map.IsPlayable, Map.CollectionName, Map.Type)) {
				NewCategory.Playlist.add(Map);
			}
		}

		if (NewCategory.Playlist.count > 0) {
			Campaign.Categories.add(NewCategory);
		}
	}

	return Campaign;
}

CampaignStruct::LibCampaignStruct_K_MonthlyCampaign FilterCampaignMaps(K_Settings _Settings, CampaignStruct::LibCampaignStruct_K_MonthlyCampaign _MonthlyCampaign) {
	if (!MapsShouldBeFiltered(_Settings)) return _MonthlyCampaign;

	declare CampaignStruct::LibCampaignStruct_K_MonthlyCampaign MonthlyCampaign = _MonthlyCampaign;
	MonthlyCampaign.DailyMaps = [];

	foreach (DayIndex => DailyMap in _MonthlyCampaign.DailyMaps) {
		if (FilterMap(_Settings, DailyMap.Map.AuthorLogin, DailyMap.Map.IsPlayable, DailyMap.Map.CollectionName, DailyMap.Map.Type)) {
			MonthlyCampaign.DailyMaps[DayIndex] = DailyMap;
		} else {
			MonthlyCampaign.DailyMaps[DayIndex] = CampaignStruct::LibCampaignStruct_K_DailyMap {};
		}
	}

	return MonthlyCampaign;
}

Void Focus(CMlScript _This, CMlControl _To) {
	if (Transition::IsInTransition(_This)) return;
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;

	declare Focused_Control = Navigation::GetFocusedControl(_This, C_NavGroupName);
	if (Focused_Control != Null) {
		Navigation::Unfocus(_This, Focused_Control);
		if (Focused_Control is CMlFrame) {
			FocusOption((Focused_Control as CMlFrame), False);
			Button::Unfocus(_This, (Focused_Control as CMlFrame));
			if (Focused_Control.HasClass(C_Class_Header)) {
				Trackmania_BrowserAPI_State.Controls.Frame_FocusHeader = Focused_Control;
			} else if (Focused_Control.HasClass(C_Class_Body)) {
				Trackmania_BrowserAPI_State.Controls.Frame_FocusBody = Focused_Control;
			} else if (Focused_Control.HasClass(C_Class_Footer)) {
				Trackmania_BrowserAPI_State.Controls.Frame_FocusFooter = Focused_Control;
			}
		}
	}

	if (_To != Null) {
		Navigation::Focus(_This, _To);
		if (_To is CMlFrame) {
			FocusOption((_To as CMlFrame), True);
			Button::Focus(_This, (_To as CMlFrame));
			if (Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll != Null && _To.ControlId == Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll.ControlId && !IsScrollFocused(_This)) {
				FocusScroll(_This, True);
			}
			Trackmania_BrowserAPI_State.PendingEvents.add(K_Event {
				Type = C_EventType_Focus
			});
			Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].IsFocused = True;
		}
	}
}

Integer GetBrowserDataCount(CMlScript _This, Integer _BrowserType) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (!Trackmania_BrowserAPI_State.Browsers.existskey(_BrowserType)) return C_Browser_NullDataCount;
	declare Integer DataCount = C_Browser_NullDataCount;

	switch (Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath) {
		case C_Path_Base: {
			if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Map) || Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Campaign)) {
				if (Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent.existskey(CampaignStruct::C_CampaignType_Quarterly)) {
					if (Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent[CampaignStruct::C_CampaignType_Quarterly]) DataCount += 1;
				} else if (Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent.existskey(CampaignStruct::C_CampaignType_Monthly)) {
					if (Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent[CampaignStruct::C_CampaignType_Monthly]) DataCount += 1;
				}
				if (Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent.existskey(CampaignStruct::C_CampaignType_Club)) {
					if (Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent[CampaignStruct::C_CampaignType_Club]) DataCount += 1;
				}
			}
			if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].Settings.ShowFavoriteUserMaps) DataCount += 1;
			return DataCount+1;
		}
		case C_Path_Officials: {
			if (Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent.existskey(CampaignStruct::C_CampaignType_Quarterly)) {
				if (Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent[CampaignStruct::C_CampaignType_Quarterly]) DataCount += 1;
			}
			if (Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent.existskey(CampaignStruct::C_CampaignType_Monthly)) {
				if (Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent[CampaignStruct::C_CampaignType_Monthly]) DataCount += 1;
			}
			return DataCount;
		}
		case C_Path_Clubs: {
			return ClubStore::GetBrowserClubsTotal();
		}
		case C_Path_Quarterly: {
			if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenCampaign.Id != CampaignStruct::C_Campaign_NullId) {
				foreach (Category in Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenCampaign.Categories) {
					DataCount += Category.Playlist.count;
				}
				return DataCount;
			} else {
				return CampaignStore::GetCampaignsTotal(CampaignStruct::C_CampaignType_Quarterly);
			}
		}
		case C_Path_Monthly: {
			if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenMonthlyCampaign.Id != CampaignStruct::C_Campaign_NullId) {
				return Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenMonthlyCampaign.DailyMaps.count;
			} else {
				return CampaignStore::GetMonthlyCampaignsTotal();
			}
		}
		case C_Path_ChosenClub: {
			if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenCampaign.Id != CampaignStruct::C_Campaign_NullId) {
				foreach (Category in Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenCampaign.Categories) {
					DataCount += Category.Playlist.count;
				}
				return DataCount;
			} else {
				return CampaignStore::GetClubCampaignsTotal(Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenClubId);
			}
		}
		case C_Path_Favorites: {
			return MapStore::GetFavoriteMapsTotal();
		}
		default: {
			foreach (Type in Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes) {
				if (C_DataType_IsFileList.exists(Type) && Trackmania_BrowserAPI_State.Browsers[_BrowserType].Files.existskey(Type)) {
					DataCount += Trackmania_BrowserAPI_State.Browsers[_BrowserType].Files[Type].Datas.count;
				} else {
					switch (Type) {
						case C_DataType_Map: {
							DataCount += Trackmania_BrowserAPI_State.Browsers[_BrowserType].Maps.count;
						}
						case C_DataType_Replay: {
							DataCount += Trackmania_BrowserAPI_State.Browsers[_BrowserType].Replays.count;
						}
						case C_DataType_Campaign: {
							DataCount += Trackmania_BrowserAPI_State.Browsers[_BrowserType].Campaigns.count;
						}
					}
				}
			}
		}
	}
	return DataCount;
}

Integer GetDataType(CMlScript _This, Integer _BrowserType, Integer _DataKey) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (!Trackmania_BrowserAPI_State.Browsers.existskey(_BrowserType)) return C_DataType_None;
	declare CurrentBrowser = Trackmania_BrowserAPI_State.Browsers[_BrowserType];

	switch (CurrentBrowser.CurrentPath) {
		case C_Path_Base: return C_DataType_Category;
		case C_Path_Officials: return C_DataType_Category;
		case C_Path_Clubs: return C_DataType_Club;
		case C_Path_Quarterly: {
			if (CurrentBrowser.ChosenCampaign.Id != CampaignStruct::C_Campaign_NullId) {
				return C_DataType_Map;
			} else {
				return C_DataType_Campaign;
			}
		}
		case C_Path_Monthly: {
			if (CurrentBrowser.ChosenMonthlyCampaign.Id != CampaignStruct::C_Campaign_NullId) {
				return C_DataType_Map;
			} else {
				return C_DataType_MonthlyCampaign;
			}
		}
		case C_Path_ChosenClub: {
			if (CurrentBrowser.ChosenCampaign.Id != CampaignStruct::C_Campaign_NullId) {
				return C_DataType_Map;
			} else {
				return C_DataType_Campaign;
			}
		}
		case C_Path_Favorites: {
			return C_DataType_Map;
		}
		default: {
			declare DataCount = 0;
			foreach (Type in CurrentBrowser.DataTypes) {
				if (C_DataType_IsFileList.exists(Type) && CurrentBrowser.Files.existskey(Type)) {
					if (CurrentBrowser.Files[Type].Datas.count + DataCount > _DataKey) return Type;
					DataCount += CurrentBrowser.Files[Type].Datas.count;
				} else {
					switch (Type) {
						case C_DataType_Map: {
							if (CurrentBrowser.Maps.count + DataCount > _DataKey) return Type;
							DataCount += CurrentBrowser.Maps.count;
						}
						case C_DataType_Replay: {
							if (CurrentBrowser.Replays.count + DataCount > _DataKey) return Type;
							DataCount += CurrentBrowser.Replays.count;
						}
						case C_DataType_Campaign: {
							if (CurrentBrowser.Campaigns.count + DataCount > _DataKey) return Type;
							DataCount += CurrentBrowser.Campaigns.count;
						}
					}
				}
			}
		}
	}
	return C_DataType_None;
}

	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (_Frame_Scroll == Null) return;

	Navigation::Lock(_Frame_Scroll, _DataCount <= 0);
	if (IsScrollFocused(_This)) {
		if (_DataCount <= 0) {
			declare Frame_ParentPath <=> (Trackmania_BrowserAPI_State.Controls.Frame_Global.GetFirstChild(C_Id_FrameParentPath) as CMlFrame);
			Focus(_This, Frame_ParentPath);
			FocusScroll(_This, False);
		} else {
			Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].IsFocused = True;
			FocusScroll(_This, True);
		}
	}
	Scroll::SetElementsTotal(_This, _Frame_Scroll, _DataCount);
}

}

Void FetchLiveData(CMlScript _This, Integer _BrowserType) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (!Trackmania_BrowserAPI_State.Browsers.existskey(_BrowserType)) return;

	LoadingSpinner::StartAnimation(_This, Trackmania_BrowserAPI_State.Controls.Frame_Global.GetFirstChild(C_Id_FrameLoadingSpinner));
	if (Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll != Null) {
		UpdateBrowserContent(_This, Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll, 0); //< Avoid some flickering
		Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll.Visible = False;
	}

	declare ElementsPerPage = GetElementsPerPage(_This, _BrowserType);
	declare Integer Length = ElementsPerPage * 3;
	declare Integer QuaterlyCampaignsOffset = ML::Max(0, (Trackmania_BrowserAPI_State.RequestsManager.PageQuarterly - 2)) * ElementsPerPage;
	CampaignStore::Action_LoadCampaigns(_This, CampaignStruct::C_CampaignType_Quarterly, QuaterlyCampaignsOffset, Length);
	CampaignStore::Action_LoadMonthlyCampaignsPage(_This, Trackmania_BrowserAPI_State.RequestsManager.PageMonthly, ElementsPerPage);
	declare Integer ClubsOffset = ML::Max(0, (Trackmania_BrowserAPI_State.RequestsManager.PageClubs - 2)) * ElementsPerPage;
	ClubStore::Action_LoadClubsBrowser(_This, ClubsOffset, Length);
	Trackmania_BrowserAPI_State.RequestsManager.WaitingForMonthlyCampaigns = True;
	Trackmania_BrowserAPI_State.RequestsManager.WaitingForQuarterlyCampaigns = True;
	Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubs = True;

Void FetchLocalData(CMlScript _This, Integer _BrowserType, Text _Path) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (!Trackmania_BrowserAPI_State.Browsers.existskey(_BrowserType)) return;
	declare K_Browser CurrentBrowser = Trackmania_BrowserAPI_State.Browsers[_BrowserType];

	declare Integer BrowserFilter = Const::C_BrowserFilter_UserAndTitleData;
	if (Config::HasFeature(Config::C_Feature_DebugBrowserAllData)) BrowserFilter = Const::C_BrowserFilter_AllData;
	foreach (Type in CurrentBrowser.DataTypes) {
		if (Type == C_DataType_Map) {
			if (CurrentBrowser.Settings.ShowFavoriteUserMaps && CurrentBrowser.CurrentPath == C_Path_Favorites) {
				Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMaps = True;
				Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
			} else {
				if (CurrentBrowser.TaskId_GetMaps != NullId) {
					DataFileMgr.TaskResult_Release(CurrentBrowser.TaskId_GetMaps);
					Trackmania_BrowserAPI_State.Browsers[_BrowserType].TaskId_GetMaps = NullId;
				}
				declare CTaskResult_MapList Task = DataFileMgr.Map_GetFilteredGameList(
					BrowserFilter,
					_Path,
					CurrentBrowser.IsFlatten,
					CurrentBrowser.IsSortingByNameElseByDate,
					CurrentBrowser.IsSortingFromAscendingOrder
				);
				if (Task != Null) {
					Trackmania_BrowserAPI_State.Browsers[_BrowserType].TaskId_GetMaps = Task.Id;
				}
			}
		} else if (Type == C_DataType_Replay) {
			if (CurrentBrowser.TaskId_GetReplays != NullId) {
				DataFileMgr.TaskResult_Release(CurrentBrowser.TaskId_GetReplays);
				Trackmania_BrowserAPI_State.Browsers[_BrowserType].TaskId_GetReplays = NullId;
			}
			declare CTaskResult_ReplayList Task = DataFileMgr.Replay_GetFilteredGameList(
				BrowserFilter,
				_Path,
				CurrentBrowser.IsFlatten,
				CurrentBrowser.IsSortingByNameElseByDate,
				CurrentBrowser.IsSortingFromAscendingOrder
			);
			if (Task != Null) {
				Trackmania_BrowserAPI_State.Browsers[_BrowserType].TaskId_GetReplays = Task.Id;
			}
		} else if (CurrentBrowser.Files.existskey(Type)) {
			if (CurrentBrowser.Files[Type].TaskId_GetFiles != NullId) {
				DataFileMgr.TaskResult_Release(CurrentBrowser.Files[Type].TaskId_GetFiles);
				Trackmania_BrowserAPI_State.Browsers[_BrowserType].Files[Type].TaskId_GetFiles = NullId;
			}
			declare CDataFileMgr::EMediaType MediaType = GetMediaType(Type);
			if (Type == C_DataType_Script) BrowserFilter = Const::C_BrowserFilter_AllData;
			declare CTaskResult_FileList Task = DataFileMgr.Media_GetFilteredGameList(
				MediaType,
				BrowserFilter,
				_Path,
				CurrentBrowser.IsFlatten,
				CurrentBrowser.IsSortingByNameElseByDate,
				CurrentBrowser.IsSortingFromAscendingOrder
			);
			if (Task != Null) {
				Trackmania_BrowserAPI_State.Browsers[_BrowserType].Files[Type].TaskId_GetFiles = Task.Id;
			}
		}
	}
}

Void FetchLocalData(CMlScript _This, Integer _BrowserType) {
	FetchLocalData(_This, _BrowserType, GetFullPath(_This, _BrowserType));
}

Void SetParentPathToCurrentPath(CMlScript _This, Integer _BrowserType) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (
		!Trackmania_BrowserAPI_State.Browsers.existskey(_BrowserType) ||
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath == C_Path_Base
	) return;

	declare NewCurrentPath = C_Path_Null;

	foreach (ParentPath => PathList in Trackmania_BrowserAPI_State.Paths_Tree) {
		if (PathList.exists(Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath)) {
			NewCurrentPath = ParentPath;
			break;
		}
	}
	foreach (ParentPath => PathList in Trackmania_BrowserAPI_State.Paths_Tree) {
		if (PathList.exists(NewCurrentPath)) {
			Trackmania_BrowserAPI_State.Browsers[_BrowserType].ParentPath = ParentPath;
			break;
		}
	}

	if (NewCurrentPath == C_Path_Null) {
		NewCurrentPath = Trackmania_BrowserAPI_State.Browsers[_BrowserType].ParentPath;

		// Can only go back to C_Path_Base (official, club, local) if there are maps or campaigns in the data
		if (
			NewCurrentPath == C_Path_Base &&
			!Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Map) &&
			!Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Campaign)
		) {
			NewCurrentPath = "";
		}

		Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath = NewCurrentPath;

		if (NewCurrentPath == C_Path_Base) {
			if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Map) || Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Campaign)) {
				FetchLiveData(_This, _BrowserType);
			}
		} else if (NewCurrentPath == "") {
			Trackmania_BrowserAPI_State.Browsers[_BrowserType].ParentPath = C_Path_Base;
			FetchLocalData(_This, _BrowserType);
		} else {
			declare Text CurrentPath = TL::Replace(Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath, "\\", "/");
			declare Text[] TmpSplit = TL::Split("/", CurrentPath);
			declare Rem = TmpSplit.removekey(TmpSplit.count-1);
			Trackmania_BrowserAPI_State.Browsers[_BrowserType].ParentPath = TL::Join("/", TmpSplit);
			FetchLocalData(_This, _BrowserType);
		}

		if (_BrowserType == Trackmania_BrowserAPI_State.Type_CurrentBrowser) {
			SetPathLabelValue(_This, _BrowserType, Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath);
		}
	} else {
		if (NewCurrentPath == C_Path_Monthly) {
			Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenMonthlyCampaign = CampaignStruct::LibCampaignStruct_K_MonthlyCampaign {
				Id = CampaignStruct::C_Campaign_NullId
			};
		} else if (NewCurrentPath == C_Path_Quarterly) {
			Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenCampaign = CampaignStruct::LibCampaignStruct_K_Campaign {
				Id = CampaignStruct::C_Campaign_NullId
			};
		} else if (NewCurrentPath == C_Path_Clubs) {
			Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenClubId = ClubStruct::C_ClubNullId;
		}

		Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath = NewCurrentPath;
		SetPathLabelValue(_This, _BrowserType, Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath);

		UpdateBrowserContent(_This, Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll, GetBrowserDataCount(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser));
	}
}

Void RefreshBrowserData(CMlScript _This) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (!Trackmania_BrowserAPI_State.Browsers.existskey(Trackmania_BrowserAPI_State.Type_CurrentBrowser)) return;
	declare K_Browser CurrentBrowser = Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser];

	declare Integer BrowserFilter = Const::C_BrowserFilter_UserAndTitleData;
	if (Config::HasFeature(Config::C_Feature_DebugBrowserAllData)) BrowserFilter = Const::C_BrowserFilter_AllData;

	foreach (Type in CurrentBrowser.DataTypes) {
		if (Type == C_DataType_Map) {
			DataFileMgr.Map_RefreshFromDisk();
			if (CurrentBrowser.Settings.ShowFavoriteUserMaps && CurrentBrowser.CurrentPath == C_Path_Favorites) {
				Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMaps = True;
				Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
			}
		} else if (Type == C_DataType_Replay) {
			DataFileMgr.Replay_RefreshFromDisk();
		} else if (CurrentBrowser.Files.existskey(Type)) {
			declare CDataFileMgr::EMediaType MediaType = GetMediaType(Type);
			if (Type == C_DataType_Script) BrowserFilter = Const::C_BrowserFilter_AllData;
			DataFileMgr.Media_RefreshFromDisk(MediaType, BrowserFilter);
		}
	}

	FetchLocalData(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
}

Void SetCurrentPath(CMlScript _This, Integer _BrowserType, Text _Path) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (!Trackmania_BrowserAPI_State.Browsers.existskey(_BrowserType)) return;

	Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath = _Path;
	foreach (ParentPath => PathList in Trackmania_BrowserAPI_State.Paths_Tree) {
		if (PathList.exists(_Path)) {
			Trackmania_BrowserAPI_State.Browsers[_BrowserType].ParentPath = ParentPath;
			break;
		}
	}

	switch (_Path) {
		case C_Path_Base: {
			if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Map) || Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Campaign)) {
				FetchLiveData(_This, _BrowserType);
			}
		}
		case C_Path_Officials: {
			if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].BasePath == C_Path_Officials) {
				Trackmania_BrowserAPI_State.RequestsManager.WaitingForMonthlyCampaigns = True;
				Trackmania_BrowserAPI_State.RequestsManager.WaitingForQuarterlyCampaigns = True;
				Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
			}
		}
		case C_Path_Monthly: {
			if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].BasePath == C_Path_Monthly) {
				Trackmania_BrowserAPI_State.RequestsManager.WaitingForMonthlyCampaigns = True;
				Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
			}
			Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenCampaign = CampaignStruct::LibCampaignStruct_K_Campaign {
				Id = CampaignStruct::C_Campaign_NullId
			};
			Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenMonthlyCampaign = CampaignStruct::LibCampaignStruct_K_MonthlyCampaign {
				Id = CampaignStruct::C_Campaign_NullId
			};
		}
		case C_Path_Quarterly: {
			if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].BasePath == C_Path_Quarterly) {
				Trackmania_BrowserAPI_State.RequestsManager.WaitingForQuarterlyCampaigns = True;
				Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
			}
			Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenMonthlyCampaign = CampaignStruct::LibCampaignStruct_K_MonthlyCampaign {
				Id = CampaignStruct::C_Campaign_NullId
			};
			Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenCampaign = CampaignStruct::LibCampaignStruct_K_Campaign {
				Id = CampaignStruct::C_Campaign_NullId
			};
		}
		case C_Path_Clubs: {
			if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].BasePath == C_Path_Clubs) {
				Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubs = True;
				Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
			}
			Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenClubId = ClubStruct::C_ClubNullId;
		}
		case C_Path_ChosenClub: {
			Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubCampaigns = True;
			Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
			Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenCampaign = CampaignStruct::LibCampaignStruct_K_Campaign {
				Id = CampaignStruct::C_Campaign_NullId
			};
		}
		case C_Path_Favorites: {
			Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMaps = True;
			Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
			ResetBrowserCampaignsAndClubId(_This, _BrowserType);
		}
		default: {
			if (_Path == "" || _Path == C_Path_Local) {
				Trackmania_BrowserAPI_State.Browsers[_BrowserType].ParentPath = C_Path_Base;
				Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath = "";
				ResetBrowserCampaignsAndClubId(_This, _BrowserType);
			} else {
				declare Text CurrentPath = TL::Replace(Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath, "\\", "/");
				declare Text[] TmpSplit = TL::Split("/", CurrentPath);
				declare Rem = TmpSplit.removekey(TmpSplit.count-1);
				Trackmania_BrowserAPI_State.Browsers[_BrowserType].ParentPath = TL::Join("/", TmpSplit);
			}
			FetchLocalData(_This, _BrowserType);
		}
	}

	if (_BrowserType == Trackmania_BrowserAPI_State.Type_CurrentBrowser) {
		SetPathLabelValue(_This, _BrowserType, Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath);
	}
}

Void InitializeOptions(CMlScript _This, Integer _BrowserType) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	declare Frame_Options <=> (Trackmania_BrowserAPI_State.Controls.Frame_Global.GetFirstChild(C_Id_FrameOptions) as CMlFrame);

	declare IsSortedFromAscendingOrder = IsSortingFromAscendingOrder(_This, _BrowserType);
	declare IsSortedByNameElseByDate = IsSortingByNameElseByDate(_This, _BrowserType);
	declare IsFlatten = IsFlatten(_This, _BrowserType);

	foreach (Control in Frame_Options.Controls) {
		if (!(Control is CMlFrame)) continue;
		declare Frame_Option = (Control as CMlFrame);
		declare Quad_Image = (Frame_Option.GetFirstChild(C_Id_QuadImage) as CMlQuad);
		if (Quad_Image == Null) continue;

		switch (Frame_Option.ControlId) {
			case C_Id_FrameOptionsRefresh: {
				Quad_Image.ChangeImageUrl(Icons::C_Icon_128x128_Refresh);
			}
			case C_Id_FrameOptionsOrder: {
				Quad_Image.ChangeImageUrl(Icons::C_Icon_128x128_Order);
				if (IsSortedFromAscendingOrder) {
					Quad_Image.RelativeRotation = 180.;
				} else {
					Quad_Image.RelativeRotation = 0.;
				}
			}
			case C_Id_FrameOptionsCriterion: {
				if (IsSortedByNameElseByDate) {
					Quad_Image.ImageUrl = Icons::C_Icon_128x128_Criterion_Alpha;
				} else {
					Quad_Image.ImageUrl = Icons::C_Icon_128x128_Criterion_Date;
				}
			}
			case C_Id_FrameOptionsFlatten: {
				if (IsFlatten) {
					Quad_Image.ImageUrl = Icons::C_Icon_128x128_Flatten_Files;
				} else {
					Quad_Image.ImageUrl = Icons::C_Icon_128x128_Flatten_Arbo;
				}
			}
			case C_Id_FrameOptionsExplorer: {
				Quad_Image.ChangeImageUrl(Icons::C_Icon_128x128_Explorer);
			}
		}
	}
}

Void UpdateBrowser(CMlScript _This, Integer _BrowserType) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (!Trackmania_BrowserAPI_State.Browsers.existskey(_BrowserType)) return;

	if (
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Map) ||
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Replay) ||
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Image) ||
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Skin) ||
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Script) ||
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_MatchSettings) ||
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_ItemCollection)
	) {
		FetchLocalData(_This, _BrowserType);
	}
	if (
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Map) ||
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Campaign) ||
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_MonthlyCampaign)
	) {
		FetchLiveData(_This, _BrowserType);
	}

	SetPathLabelValue(_This, _BrowserType, Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath);
	UpdateBrowserTitle(_This, _BrowserType);
	UpdateBrowserSubTitle(_This, _BrowserType);
	InitializeOptions(_This, _BrowserType);
	UpdateBrowserContent(_This, Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll, GetBrowserDataCount(_This, _BrowserType));
}

Void CheckSettings(Integer _BrowserType, K_Settings _Settings, Integer[] _DataTypes) {
	assert(!_Settings.ListMultiSelect || !_Settings.DisplayEditorSettingsButton, "ListMultiSelect and DisplayEditorSettingsButton cannot be active at the same time");
	assert(!_Settings.ListMultiSelect || (_DataTypes.count == 1 && _DataTypes.containsonly(C_DataType_CanBeMultiSelect))); //< Multi selection works only if the browser has one data type
	assert(!_Settings.DisplayEditorSettingsButton || _BrowserType == C_BrowserType_PathNavigationSmall, "DisplayEditorSettingsButton can only be activated with browser type PathNavigationSmall");
}

Void AddBrowser(
	CMlScript _This,
	Integer _BrowserType,
	Text _BrowserName,
	Text _BrowserSubTitle,
	Integer[] _DataTypes,
	Text _BasePath,
	Text _StartPath,
	CTaskResult_MapList _TaskResultMapList,
	CTaskResult_ReplayList _TaskResultReplayList,
	CTaskResult_FileList[Integer] _TaskResultFileLists,
	Boolean _IsFlatten,
	Boolean _IsSortingFromAscendingOrder,
	Boolean _IsSortingByNameElseByDate,
	K_Settings _Settings
) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	CheckSettings(_BrowserType, _Settings, _DataTypes);

	declare K_TaskResult_FileData[Integer] Files;
	foreach (DataType in _DataTypes) {
		if (!C_DataType_IsFileList.exists(DataType)) continue;
		Files[DataType] = K_TaskResult_FileData{};
		if (!_TaskResultFileLists.existskey(DataType)) continue;
		if (_TaskResultFileLists[DataType] != Null) {
			Files[DataType].TaskId_GetFiles = _TaskResultFileLists[DataType].Id;
		} else {
			Files[DataType].TaskId_GetFiles = NullId;
		}
	}

	Trackmania_BrowserAPI_State.Browsers[_BrowserType] = K_Browser {
		Type = _BrowserType,
		Name = _BrowserName,
		SubTitle = _BrowserSubTitle,
		DataTypes = _DataTypes,
		BasePath = _BasePath,
		ParentPath = "",
		CurrentPath = "",
		ChosenClubId = ClubStruct::C_ClubNullId,
		ChosenCampaign = CampaignStruct::LibCampaignStruct_K_Campaign {
			Id = CampaignStruct::C_Campaign_NullId
		},
		ChosenMonthlyCampaign = CampaignStruct::LibCampaignStruct_K_MonthlyCampaign {
			Id = CampaignStruct::C_Campaign_NullId
		},
		Files = Files,
		IsFlatten = _IsFlatten,
		IsSortingFromAscendingOrder = _IsSortingFromAscendingOrder,
		IsSortingByNameElseByDate = _IsSortingByNameElseByDate,
		Settings = _Settings,
		SelectedReplays = [],
		SelectedMaps = []
	};
	SetCurrentPath(_This, _BrowserType, _StartPath);
	UpdateBrowserTitle(_This, _BrowserType);
	UpdateBrowserSubTitle(_This, _BrowserType);
	InitializeOptions(_This, _BrowserType);
}

Void AddBrowser(CMlScript _This, Integer _BrowserType, Text _BrowserName, Text _BrowserSubTitle, Integer[] _DataTypes, Text _StartPath, K_Settings _Settings) {
	AddBrowser(_This, _BrowserType, _BrowserName, _BrowserSubTitle, _DataTypes, "", _StartPath, Null, Null, [], False, False, False, _Settings);
}

CampaignStruct::LibCampaignStruct_K_Campaign GetCampaign(CMlScript _This, Integer _BrowserType, Integer _DataKey) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (!Trackmania_BrowserAPI_State.Browsers.existskey(_BrowserType)) {
		return CampaignStruct::LibCampaignStruct_K_Campaign {
			Id = CampaignStruct::C_Campaign_NullId
		};
	}

	declare Campaign = CampaignStruct::LibCampaignStruct_K_Campaign {
		Id = CampaignStruct::C_Campaign_NullId
	};

	switch (Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath) {
		case C_Path_Quarterly: {
			declare CampaignList = CampaignStore::GetCampaignList(CampaignStruct::C_CampaignType_Quarterly);
			if (CampaignList.existskey(_DataKey)) {
				Campaign = CampaignList[_DataKey];
				if (Campaign.IsLoaded) {
					Campaign = CampaignStore::GetCampaignFromId(Campaign.Id);
				}
			}
		}
		case C_Path_ChosenClub: {
			declare CampaignList = CampaignStore::GetClubCampaignList(Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenClubId);
			if (CampaignList.existskey(_DataKey)) {
				Campaign = CampaignList[_DataKey];
				if (Campaign.IsLoaded) {
					Campaign = CampaignStore::GetCampaignFromId(Campaign.Id);
				}
			}
		}
		default: {
			if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].Campaigns.existskey(_DataKey)) {
				return Trackmania_BrowserAPI_State.Browsers[_BrowserType].Campaigns[_DataKey];
			}
		}
	}

	return Campaign;
}

CampaignStruct::LibCampaignStruct_K_MonthlyCampaign GetMonthlyCampaign(Integer _DataKey) {
	declare MonthlyCampaignList = CampaignStore::GetMonthlyCampaignList();
	declare MonthlyCampaign = CampaignStruct::LibCampaignStruct_K_MonthlyCampaign {
		Id = CampaignStruct::C_Campaign_NullId
	};

	if (MonthlyCampaignList.existskey(_DataKey)) {
		MonthlyCampaign = CampaignStore::GetMonthlyCampaignFromId(MonthlyCampaignList[_DataKey].Id);
		if (!MonthlyCampaign.IsLoaded) {
			MonthlyCampaign = MonthlyCampaignList[_DataKey];
		}
	}
	return MonthlyCampaign;
}

K_MapBrowserData GetMapBrowserData(CMlScript _This, Integer _BrowserType, Integer _DataKey) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (!Trackmania_BrowserAPI_State.Browsers.existskey(_BrowserType) || !Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Map)) return K_MapBrowserData{};

	if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenCampaign.Id != CampaignStruct::C_Campaign_NullId) {
		declare MapCount = 0;
		declare CampaignStruct::LibCampaignStruct_K_Map ChosenMap;
		foreach (Category in Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenCampaign.Categories) {
			if (Category.Playlist.count + MapCount <= _DataKey) {
				MapCount += Category.Playlist.count;
				continue;
			} else {
				if (Category.Playlist.existskey(_DataKey - MapCount)) ChosenMap = Category.Playlist[_DataKey - MapCount];
				if (ChosenMap.Uid != "") break;
			}
		}

		return K_MapBrowserData{
			MapInfo = ChosenMap
		};
	} else if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenMonthlyCampaign.Id != CampaignStruct::C_Campaign_NullId) {
		// DailyMaps have a day index and not a key like other map arrays
		declare Integer Key;
		foreach (DayIndex => DailyMap in Trackmania_BrowserAPI_State.Browsers[_BrowserType].ChosenMonthlyCampaign.DailyMaps) {
			if (Key == _DataKey) {
				return K_MapBrowserData {
					MapInfo = DailyMap.Map
				};
				break;
			}
			Key += 1;
		}
	} else if (
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].Settings.ShowFavoriteUserMaps &&
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].CurrentPath == C_Path_Favorites &&
		Trackmania_BrowserAPI_State.Browsers[_BrowserType].FavoriteMaps.existskey(_DataKey)
	) {
		return Trackmania_BrowserAPI_State.Browsers[_BrowserType].FavoriteMaps[_DataKey];
	} else if (Trackmania_BrowserAPI_State.Browsers[_BrowserType].Maps.existskey(_DataKey)) {
		return Trackmania_BrowserAPI_State.Browsers[_BrowserType].Maps[_DataKey];
	}

	return K_MapBrowserData {};
}

	);

	Frame_SelectedSkin.Visible = Frame_BrowserTestMapScroll.Visible && Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Skin);

	Navigation::ChangeControlTarget(_This, Frame_ButtonCreateMap, "up", Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll.ControlId);
	Navigation::ChangeControlTarget(_This, Frame_ButtonSelectMap, "up", Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll.ControlId);
	Navigation::ChangeControlTarget(_This, Button_EditorSettings, "left", Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll.ControlId);
	Navigation::ChangeControlTarget(_This, Button_ListMultiSelectValid, "up", Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll.ControlId);
	Navigation::ChangeControlTarget(_This, Button_ListMultiSelectSelectAll, "down", Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll.ControlId);
	Navigation::ChangeControlTarget(_This, Button_ListMultiSelectunselectAll, "down", Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll.ControlId);
	if (Trackmania_BrowserAPI_State.Browsers.existskey(_BrowserType)) {
		Frame_ButtonCreateMap.Visible = Trackmania_BrowserAPI_State.Browsers[_BrowserType].Settings.DisplayCreateMapButton && Trackmania_BrowserAPI_State.Browsers[_BrowserType].DataTypes.exists(C_DataType_Map);
		Navigation::Lock(Frame_ButtonCreateMap, !Frame_ButtonCreateMap.Visible);
		Frame_ButtonSelectMap.Hide();
		Navigation::Lock(Frame_ButtonSelectMap, True);
		Button_EditorSettings.Visible = False;
		Navigation::Lock(Button_EditorSettings, True);
		UpdateMapPreselection(_This, K_MapBrowserData {}, Frame_SelectedMap);
		Frame_Tooltip.Visible = Trackmania_BrowserAPI_State.Browsers[_BrowserType].Settings.DisplayFileNameTooltip;
		if (Frame_Tooltip.Visible) {
			switch (Trackmania_BrowserAPI_State.Browsers[_BrowserType].Settings.PositionFileNameTooltip) {
				case C_TooltipPos_ScrollTop: {
					Frame_Tooltip.RelativePosition_V3 = <4.5, 22.>;
					Label_Tooltip.TextSizeReal = 4.;
				}
				default: { // C_TooltipPos_Footer
					Frame_Tooltip.RelativePosition_V3 = <-12., -69.>;
					Label_Tooltip.TextSizeReal = 5.;
				}
			}
		}

		Frame_ListButtons.Visible = (Frame_BrowserListScroll.Visible || Frame_BrowserTestMapScroll.Visible) && Trackmania_BrowserAPI_State.Browsers[_BrowserType].Settings.ListMultiSelect;
		Navigation::Lock(Button_ListMultiSelectValid, True);
		Navigation::Lock(Button_ListMultiSelectSelectAll, !Frame_ListButtons.Visible);
		Navigation::Lock(Button_ListMultiSelectunselectAll, !Frame_ListButtons.Visible);
		UpdateMultiSelectButtons(_This, Button_ListMultiSelectValid);
		UpdateMultiSelectButtons(_This, Frame_ButtonSelectMap);

		Frame_ButtonCreateMap.RelativePosition_V3 = <45., -68.5>;
		if (Frame_ListButtons.Visible) {
			Navigation::ChangeControlTarget(_This, Frame_ButtonCreateMap, "right", Button_ListMultiSelectValid.ControlId);
		} else if (Frame_SelectedMap.Visible) {
			Navigation::ChangeControlTarget(_This, Frame_ButtonCreateMap, "right", Frame_ButtonSelectMap.ControlId);
		} else {
			Frame_ButtonCreateMap.RelativePosition_V3 = <93.5, -68.5>;
		}
	}
}

Void UpdateBrowserElement(CMlScript _This, CMlFrame _Frame_Element, Integer _DataKey) {
	if (_Frame_Element == Null) return;
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	declare DataTypeElement = GetDataType(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, _DataKey);
	declare K_BrowserElement BrowserElementData;
	if (C_DataType_NameList.existskey(DataTypeElement)) BrowserElementData.TypeName = C_DataType_NameList[DataTypeElement];

	declare Boolean IsSubfolder for _Frame_Element;
	switch (DataTypeElement) {
		case C_DataType_Campaign: {
			declare Campaign = GetCampaign(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, _DataKey);
			BrowserElementData.Name = Campaign.Name;
			declare Text[Integer] MissingConditions;
			declare MapNumber = 0;
			declare UnplayableMap = 0;
			foreach (Category in Campaign.Categories) {
				foreach (Map in Category.Playlist) {
					if (Map.Name != ""){
						MapNumber += 1;
						if (!Map.IsPlayable) UnplayableMap += 1;
					}
				}
			}
			if (MapNumber < 1) MissingConditions[0] = C_Text_Map;
			if (UnplayableMap >= 1) {
				declare Text Validation;
				if (UnplayableMap > 1){
					Validation = TL::GetTranslatedText(C_Text_Validations);
				} else {
					Validation = TL::GetTranslatedText(C_Text_Validation);
				}
				MissingConditions[1] = (UnplayableMap^" "^Validation);
			}
			if (Campaign.Name == "") MissingConditions[2] = C_Text_Name;
			if (MapNumber < 1 || UnplayableMap >= 1 || Campaign.Name == "") {
				BrowserElementData.Missing = TL::Compose(C_Text_Missing, MissingConditions.get(0, ""), MissingConditions.get(1, ""), MissingConditions.get(2, ""), MissingConditions.get(3, ""));
			}
		}
		case C_DataType_MonthlyCampaign: {
			declare MonthlyCampaign = GetMonthlyCampaign(_DataKey);
			if (MonthlyCampaign.Id != CampaignStruct::C_Campaign_NullId) {
				BrowserElementData.Name = TL::Compose(C_Month_Names[MonthlyCampaign.Month-1], TL::ToText(MonthlyCampaign.Year));
			} else {
				BrowserElementData.Name = Const::C_Text_RetrievingDataPlaceholder;
			}
		}
		case C_DataType_Map: {
			declare K_MapBrowserData MapBrowserData = GetMapBrowserData(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, _DataKey);
			if (MapBrowserData.SubFolder != "") {
				BrowserElementData.Name = GetLastNameOfPath(MapBrowserData.SubFolder);
				BrowserElementData.IsSubfolder = True;
			} else {
				BrowserElementData.Name = MapBrowserData.MapInfo.Name;
				BrowserElementData.MapUid = MapBrowserData.MapInfo.Uid;
			}
		}
		case C_DataType_Category: {
			BrowserElementData.IsSubfolder = True;
			declare CurrentPath = Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].CurrentPath;
			if (Trackmania_BrowserAPI_State.Paths_Tree.existskey(CurrentPath) && Trackmania_BrowserAPI_State.Paths_Tree[CurrentPath].existskey(_DataKey)) {
				declare Path = Trackmania_BrowserAPI_State.Paths_Tree[CurrentPath][_DataKey];
				if (C_Paths_Names.existskey(Path)) BrowserElementData.Name = C_Paths_Names[Path];
			}
		}
		case C_DataType_Club: {
			declare Club = GetClub(Trackmania_BrowserAPI_State.Type_CurrentBrowser, _DataKey);
			BrowserElementData.IsSubfolder = True;
			BrowserElementData.Name = Club.Name;
			BrowserElementData.File = Club.LogoUrl;
		}
		case C_DataType_Replay: {
			declare K_ReplayBrowserData ReplayBrowserData = GetReplayBrowserData(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, _DataKey);
			if (ReplayBrowserData.SubFolder != "") {
				BrowserElementData.Name = GetLastNameOfPath(ReplayBrowserData.SubFolder);
				BrowserElementData.IsSubfolder = True;
			} else {
				BrowserElementData.Name = TL::StripFormatting(ReplayBrowserData.ReplayInfo.Name);
			}
			BrowserElementData.File = ReplayBrowserData.ReplayInfo.FileName;
		}
		default: {
			if (C_DataType_IsFileList.exists(DataTypeElement)) {
				declare FileBrowserData = GetFileBrowserData(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, _DataKey);
				if (FileBrowserData.SubFolder != "") {
					BrowserElementData.IsSubfolder = True;
					BrowserElementData.Name = GetLastNameOfPath(FileBrowserData.SubFolder);
				} else {
					BrowserElementData.Name = GetLastNameOfPath(FileBrowserData.File);
					BrowserElementData.File = FileBrowserData.File;
				}
			}
		}
	}

	IsSubfolder = BrowserElementData.IsSubfolder;
	if (IsSubfolder) MenuSounds::SetCustomSelectSound(_Frame_Element, "BrowserFolderDive");
	else MenuSounds::SetCustomSelectSound(_Frame_Element, MenuSounds::C_Sound_Select);

	declare Text Trackmania_BrowserAPI_Tooltip for _Frame_Element;
	declare Text Trackmania_BrowserAPI_FileName for _Frame_Element;
	declare Text Trackmania_BrowserAPI_FilePath for _Frame_Element;
	Trackmania_BrowserAPI_Tooltip = "";
	Trackmania_BrowserAPI_FileName = "";
	Trackmania_BrowserAPI_FilePath = "";
	if (!BrowserElementData.IsSubfolder) {
		Trackmania_BrowserAPI_Tooltip = BrowserElementData.Name;
		Trackmania_BrowserAPI_FileName = BrowserElementData.Name;
		Trackmania_BrowserAPI_FilePath = BrowserElementData.File;
	}

	switch (Trackmania_BrowserAPI_State.Type_CurrentBrowser) {
		case C_BrowserType_List: {
			declare Quad_Check <=> (_Frame_Element.GetFirstChild(C_Id_QuadCheck) as CMlQuad);
			declare Label_Name <=> (_Frame_Element.GetFirstChild(C_Id_LabelName) as CMlLabel);
			declare Quad_BG <=> (_Frame_Element.GetFirstChild(C_Id_QuadBg) as CMlQuad);
			Label_Name.Value = BrowserElementData.Name;
			Label_Name.Visible = True;

			if (BrowserElementData.IsSubfolder) {
				Quad_BG.Colorize = Stylesheet::GetColorRGB(Stylesheet::C_Color_BGBrowserFolder);
				Quad_Check.Visible = False;
			} else {
				Quad_BG.Colorize = Stylesheet::GetColorRGB(Stylesheet::C_Color_AccentOne);
				Quad_Check.Visible = (
					Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.ListMultiSelect &&
					C_DataType_CanBeMultiSelect.exists(DataTypeElement)
				);
			}

			Label_Name.TextColor = Stylesheet::GetColorRGB(Stylesheet::C_Color_AccentFour);
			Label_Name.RelativePosition_V3.X = Trackmania_BrowserAPI_State.ManialinkProps.SizeX_ListMargin;
			Label_Name.Size.X = Trackmania_BrowserAPI_State.ManialinkProps.Size_BrowserLists.X - Trackmania_BrowserAPI_State.ManialinkProps.SizeX_ListMargin - Quad_Check.Size.X;

			if (Quad_Check.Visible) {
				if (
					(Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Replay) && Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedReplays.existskey(BrowserElementData.File)) ||
					(Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Map) && Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedMaps.existskey(BrowserElementData.MapUid))
				) {
					Quad_Check.ChangeImageUrl(Icons::C_Icon_128x128_Checkbox_Active_Square);
				} else {
					Quad_Check.ChangeImageUrl(Icons::C_Icon_128x128_Checkbox_Inactive_Square);
				}
			}
		}
		default: {
			if (Trackmania_BrowserAPI_State.Type_CurrentBrowser == C_BrowserType_PathNavigation || Trackmania_BrowserAPI_State.Type_CurrentBrowser == C_BrowserType_PathNavigationSmall) {
				declare Label_Name <=> (_Frame_Element.GetFirstChild(C_Id_LabelName) as CMlLabel);
				declare Quad_Image <=> (_Frame_Element.GetFirstChild(C_Id_QuadImage) as CMlQuad);
				declare Quad_BG <=> (_Frame_Element.GetFirstChild(C_Id_QuadBg) as CMlQuad);
				declare Quad_Focus <=> (_Frame_Element.GetFirstChild(C_Id_QuadFocus) as CMlQuad);
				declare Quad_Check <=> (_Frame_Element.GetFirstChild(C_Id_QuadCheck) as CMlQuad);

				Label_Name.Value = BrowserElementData.Name;
				if (Quad_Check != Null) Quad_Check.Visible = False;
				Label_Name.Visible = True;

				if (BrowserElementData.IsSubfolder) {
					Quad_BG.ChangeImageUrl(Const::C_ImageUrl_ButtonBG_Folder);
					Quad_Focus.ChangeImageUrl(Const::C_ImageUrl_ButtonBG_Folder);
					Quad_BG.Colorize = Stylesheet::GetColorRGB(Stylesheet::C_Color_BGBrowserFolder);
					Label_Name.TextColor = Stylesheet::GetColorRGB(Stylesheet::C_Color_AccentFour);
				}	else {
					Quad_BG.ChangeImageUrl(Const::C_ImageUrl_ButtonBG_File);
					Quad_Focus.ChangeImageUrl(Const::C_ImageUrl_ButtonBG_File);
					Quad_BG.Colorize = Stylesheet::GetColorRGB(Stylesheet::C_Color_AccentOne);
					Label_Name.TextColor = Stylesheet::GetColorRGB(Stylesheet::C_Color_AccentFour);
				}
				switch (DataTypeElement) {
					case C_DataType_Image: {
						Label_Name.Visible = BrowserElementData.IsSubfolder;
						Quad_Image.Visible = !BrowserElementData.IsSubfolder;
						if (!BrowserElementData.IsSubfolder) {
							Quad_Image.ChangeImageUrl(C_BasePath^C_BasePath_Media^BrowserElementData.File);
						}
					}
					default: {
						Quad_Image.Visible = False;

						if (Quad_Check != Null && C_DataType_CanBeMultiSelect.exists(DataTypeElement)) {
							Quad_Check.Visible = (
								!BrowserElementData.IsSubfolder &&
								Trackmania_BrowserAPI_State.Type_CurrentBrowser == C_BrowserType_PathNavigationSmall &&
								Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.ListMultiSelect
							);
							if (Quad_Check.Visible) {
								if (
									(Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Replay) && Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedReplays.existskey(BrowserElementData.File)) ||
									(Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Map) && Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedMaps.existskey(BrowserElementData.MapUid))
								) {
									Quad_Check.ChangeImageUrl(Icons::C_Icon_128x128_Checkbox_Active_Square);
								} else {
									Quad_Check.ChangeImageUrl(Icons::C_Icon_128x128_Checkbox_Inactive_Square);
								}
							}
						}
					}
				}
			}
		}
	}
}

Void LeaveBrowser(CMlScript _This) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	DestroyScene(_This);
	Trackmania_BrowserAPI_State.RequestsManager.WaitingForMonthlyCampaigns = False;
	Trackmania_BrowserAPI_State.RequestsManager.WaitingForQuarterlyCampaigns = False;
	Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubCampaigns = False;
	Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubs = False;
	Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedCampaign = False;
	Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedMonthlyCampaign = False;
	Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMaps = False;
	Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMapSupression = False;
	Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = -1;

	LoadingSpinner::StopAnimation(_This, Trackmania_BrowserAPI_State.Controls.Frame_Global.GetFirstChild(C_Id_FrameLoadingSpinner));

	if (!Trackmania_BrowserAPI_State.Browsers.existskey(Trackmania_BrowserAPI_State.Type_CurrentBrowser)) return;
	Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign = CampaignStruct::LibCampaignStruct_K_Campaign {
		Id = CampaignStruct::C_Campaign_NullId
	};
	Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenMonthlyCampaign = CampaignStruct::LibCampaignStruct_K_MonthlyCampaign {
		Id = CampaignStruct::C_Campaign_NullId
	};
	Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenClubId = ClubStruct::C_ClubNullId;

	Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
		Type = C_EventType_Cancel
	});
}

Void Cancel(CMlScript _This, Boolean _LeaveBrowser) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (!Trackmania_BrowserAPI_State.Browsers.existskey(Trackmania_BrowserAPI_State.Type_CurrentBrowser)) return;

	declare CurrentBrowser = Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser];
	if (CurrentBrowser.CurrentPath != C_Path_Base) {
		if (
			(!CurrentBrowser.DataTypes.containsoneof(C_DataType_HasCloudFolder) || CurrentBrowser.Settings.DisableCloudFolder) &&
			(CurrentBrowser.CurrentPath == C_Path_Null || CurrentBrowser.CurrentPath == "" || CurrentBrowser.BasePath == CurrentBrowser.CurrentPath)
		) {
			if (_LeaveBrowser) LeaveBrowser(_This);
		} else {
			SetParentPathToCurrentPath(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
		}
	} else {
		if (_LeaveBrowser) LeaveBrowser(_This);
	}
}

Void Private_UpdateListMultiSelect(CMlScript _This) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	declare CMlFrame[Integer] ElementsVisible = Scroll::GetElementsVisible(Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll);
	foreach (Key => Frame_Element in ElementsVisible) {
		if (Frame_Element.Visible) {
			UpdateBrowserElement(_This, Frame_Element, Key);
		}
	}
	declare Button_ListMultiSelectValid <=> (Trackmania_BrowserAPI_State.Controls.Frame_Global.GetFirstChild(C_Id_ButtonListMultiSelectValid) as CMlFrame);
	Navigation::Lock(Button_ListMultiSelectValid, (
		Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedReplays.count <= 0 &&
		Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedMaps.count <= 0
	));
	UpdateMultiSelectButtons(_This, Button_ListMultiSelectValid);
}

Void Select(CMlScript _This, CMlControl _Control) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (_Control == Null) return;

	switch (_Control.ControlId){
		case C_Id_FrameOptionsExplorer: {
			OpenExplorer(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
		}
		case C_Id_FrameOptionsRefresh: {
			RefreshBrowserData(_This);
		}
		case C_Id_FrameOptionsOrder: {
			declare IsSortedFromAscendingOrder = SwitchSortOrder(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
			declare Quad_Image <=> ((_Control as CMlFrame).GetFirstChild(C_Id_QuadImage) as CMlQuad);
			if (Quad_Image == Null) return;
			if (IsSortedFromAscendingOrder) {
				Quad_Image.RelativeRotation = 180.;
			} else {
				Quad_Image.RelativeRotation = 0.;
			}
		}
		case C_Id_FrameOptionsCriterion: {
			declare IsSortedByNameElseByDate = SwitchSortCriterion(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
			declare Quad_Image <=> ((_Control as CMlFrame).GetFirstChild(C_Id_QuadImage) as CMlQuad);
			if (Quad_Image == Null) return;
			if (IsSortedByNameElseByDate) {
				Quad_Image.ImageUrl = Icons::C_Icon_128x128_Criterion_Alpha;
			} else {
				Quad_Image.ImageUrl = Icons::C_Icon_128x128_Criterion_Date;
			}
		}
		case C_Id_FrameOptionsFlatten: {
			declare IsFlatten = SwitchFlatten(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
			declare Quad_Image <=> ((_Control as CMlFrame).GetFirstChild(C_Id_QuadImage) as CMlQuad);
			if (Quad_Image == Null) return;
			if (IsFlatten) {
				Quad_Image.ImageUrl = Icons::C_Icon_128x128_Flatten_Files;
			} else {
				Quad_Image.ImageUrl = Icons::C_Icon_128x128_Flatten_Arbo;
			}
		}
		case C_Id_FrameParentPath: {
			Cancel(_This, False);
		}
		case C_Id_ButtonCreateMap: {
			Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
				Type = C_EventType_CreateMap
			});
		}
		case C_Id_ButtonSelectMap: {
			Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
				Type = C_EventType_ConfirmSelection,
				DataType = C_DataType_Map,
				Data = Trackmania_BrowserAPI_State.PreSelectedMapBrowserData.MapInfo.tojson()
			});
			ResetBrowserCampaignsAndClubId(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
		}
		case C_Id_ButtonEditorSettings: {
			Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
				Type = C_EventType_EditorSettings,
				DataType = C_DataType_Map,
				Data = Trackmania_BrowserAPI_State.PreSelectedMapBrowserData.MapInfo.tojson()
			});
			ResetBrowserCampaignsAndClubId(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
		}
		case C_Id_ButtonListMultiSelectValid: {
			if (Trackmania_BrowserAPI_State.Browsers.existskey(Trackmania_BrowserAPI_State.Type_CurrentBrowser)) {
				if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Replay)) {
					declare K_ReplayInfo[] ReplayInfoList;
					foreach (ReplayInfo in Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedReplays) {
						ReplayInfoList.add(ReplayInfo);
					}
					Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
						Type = C_EventType_MultiSelect,
						DataType = C_DataType_Replay,
						Data = ReplayInfoList.tojson()
					});
				}
				if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Map)) {
					declare CampaignStruct::LibCampaignStruct_K_Map[] MapList;
					foreach (Map in Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedMaps) {
						MapList.add(Map);
					}
					Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
						Type = C_EventType_MultiSelect,
						DataType = C_DataType_Map,
						Data = MapList.tojson()
					});
				}
			}
		}
		case C_Id_ButtonListMultiSelectSelectAll: {
			if (Trackmania_BrowserAPI_State.Browsers.existskey(Trackmania_BrowserAPI_State.Type_CurrentBrowser)) {
				if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Replay)) {
					declare Text[] TrackUidsRequiringEntireCampaignAccess;
					if (
						!UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_PlayEntireOfficialQuarterlyCampaign) &&
						Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.IsPlayAgainstReplay
					) {
						TrackUidsRequiringEntireCampaignAccess = RBAC::GetTrackUidsRequiringEntireCampaignAccess(CampaignStore::GetLiveQuarterlyCampaign());
					}
					foreach (ReplayBrowserData in Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Replays) {
						if (
							ReplayBrowserData.SubFolder == "" &&
							ReplayBrowserData.ReplayInfo.FileName != "" &&
							!TrackUidsRequiringEntireCampaignAccess.exists(ReplayBrowserData.ReplayInfo.MapUid)
						) {
							Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedReplays[ReplayBrowserData.ReplayInfo.FileName] = ReplayBrowserData.ReplayInfo;
						}
					}
				}
				if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Map)) {
					if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign.Id != CampaignStruct::C_Campaign_NullId) {
						foreach (Category in Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign.Categories) {
							foreach (Map in Category.Playlist) {
								if (Map.Uid != "") {
									Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedMaps[Map.Uid] = Map;
								}
							}
						}
					} else if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenMonthlyCampaign.Id != CampaignStruct::C_Campaign_NullId) {
						foreach (DailyMap in Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenMonthlyCampaign.DailyMaps) {
							if (DailyMap.Map.Uid != "") {
								Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedMaps[DailyMap.Map.Uid] = DailyMap.Map;
							}
						}
					} else if (
						Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.ShowFavoriteUserMaps &&
						Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].CurrentPath == C_Path_Favorites
					) {
						foreach (MapBrowserData in Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].FavoriteMaps) {
							if (MapBrowserData.SubFolder == "" && MapBrowserData.MapInfo.Uid != "") {
								Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedMaps[MapBrowserData.MapInfo.Uid] = MapBrowserData.MapInfo;
							}
						}
					} else {
						foreach (MapBrowserData in Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Maps) {
							if (MapBrowserData.SubFolder == "" && MapBrowserData.MapInfo.Uid != "") {
								Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedMaps[MapBrowserData.MapInfo.Uid] = MapBrowserData.MapInfo;
							}
						}
					}
				}
				Private_UpdateListMultiSelect(_This);
			}
		}
		case C_Id_ButtonListMultiSelectUnselectAll: {
			if (Trackmania_BrowserAPI_State.Browsers.existskey(Trackmania_BrowserAPI_State.Type_CurrentBrowser)) {
				if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Replay)) {
					foreach (ReplayBrowserData in Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Replays) {
						if (ReplayBrowserData.SubFolder == "" && ReplayBrowserData.ReplayInfo.FileName != "") {
							Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedReplays.removekey(ReplayBrowserData.ReplayInfo.FileName);
						}
					}
				}
				if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Map)) {
					if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign.Id != CampaignStruct::C_Campaign_NullId) {
						foreach (Category in Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign.Categories) {
							foreach (Map in Category.Playlist) {
								if (Map.Uid != "") {
									Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedMaps.removekey(Map.Uid);
								}
							}
						}
					} else if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenMonthlyCampaign.Id != CampaignStruct::C_Campaign_NullId) {
						foreach (DailyMap in Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenMonthlyCampaign.DailyMaps) {
							if (DailyMap.Map.Uid != "") {
								Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedMaps.removekey(DailyMap.Map.Uid);
							}
						}
					} else if (
						Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.ShowFavoriteUserMaps &&
						Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].CurrentPath == C_Path_Favorites
					) {
						foreach (MapBrowserData in Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].FavoriteMaps) {
							if (MapBrowserData.SubFolder == "" && MapBrowserData.MapInfo.Uid != "") {
								Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedMaps.removekey(MapBrowserData.MapInfo.Uid);
							}
						}
					} else {
						foreach (MapBrowserData in Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Maps) {
							if (MapBrowserData.SubFolder == "" && MapBrowserData.MapInfo.Uid != "") {
								Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].SelectedMaps.removekey(MapBrowserData.MapInfo.Uid);
							}
						}
					}
				}
				Private_UpdateListMultiSelect(_This);
			}
		}
	}
}

Void InitPathsTree(CMlScript _This) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	Trackmania_BrowserAPI_State.Paths_Tree = C_Paths_Tree;
	declare Integer NumberOfOfficialsCategories = 0;
	foreach (CampaignType => IsLive in Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent) {
		switch (CampaignType) {
			case CampaignStruct::C_CampaignType_Quarterly: {
				if (IsLive) {
					NumberOfOfficialsCategories += 1;
				} else {
					foreach (ParentPath => PathList in Trackmania_BrowserAPI_State.Paths_Tree) {
						if (PathList.exists(C_Path_Quarterly)) Trackmania_BrowserAPI_State.Paths_Tree[ParentPath].remove(C_Path_Quarterly);
					}
				}
			}
			case CampaignStruct::C_CampaignType_Monthly: {
				if (IsLive) {
					NumberOfOfficialsCategories += 1;
				} else {
					foreach (ParentPath => PathList in Trackmania_BrowserAPI_State.Paths_Tree) {
						if (PathList.exists(C_Path_Monthly)) Trackmania_BrowserAPI_State.Paths_Tree[ParentPath].remove(C_Path_Monthly);
					}
				}
			}
			case CampaignStruct::C_CampaignType_Club: {
				if (!IsLive) {
					foreach (ParentPath => PathList in Trackmania_BrowserAPI_State.Paths_Tree) {
						if (PathList.exists(C_Path_Clubs)) Trackmania_BrowserAPI_State.Paths_Tree[ParentPath].remove(C_Path_Clubs);
					}
				}
			}
		}
	}
	if (NumberOfOfficialsCategories == 0) {
		foreach (ParentPath => PathList in Trackmania_BrowserAPI_State.Paths_Tree) {
			if (PathList.exists(C_Path_Officials)) Trackmania_BrowserAPI_State.Paths_Tree[ParentPath].remove(C_Path_Officials);
		}
	}
}

	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;

	} else {

}

Void Private_UpdateRequestManager(CMlScript _This) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;

	if (
		Trackmania_BrowserAPI_State.RequestsManager.WaitingForMonthlyCampaigns ||
		Trackmania_BrowserAPI_State.RequestsManager.WaitingForQuarterlyCampaigns ||
		Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubCampaigns ||
		Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubs ||
		Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedCampaign ||
		Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedMonthlyCampaign ||
		Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMaps ||
		Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMapSupression
	) {
		foreach (Event in _This.PendingEvents) {
			switch (Event.Type) {
				case CMlScriptEvent::Type::PluginCustomEvent: {
					if (
						(
							Trackmania_BrowserAPI_State.RequestsManager.WaitingForMonthlyCampaigns ||
							Trackmania_BrowserAPI_State.RequestsManager.WaitingForQuarterlyCampaigns
						) && (
							Event.CustomEventType == CampaignStore::C_Event_CampaignListUpdated ||
							Event.CustomEventType == CampaignStore::C_Event_CampaignListEmpty ||
							Event.CustomEventType == CampaignStore::C_Event_CampaignListUpdateError
						)
					) {
						if (Event.CustomEventData.count > 0) {
							// @TODO When campaigntypes are removed we need to differientiate with the new Event names and place PathType as key of Boolean Array
							declare Integer CampaignType = TL::ToInteger(Event.CustomEventData[0]);
							foreach	(BrowserType => Browser in Trackmania_BrowserAPI_State.Browsers) {
								Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent[CampaignType] = Event.CustomEventType == CampaignStore::C_Event_CampaignListUpdated;
							}
							if (CampaignType == CampaignStruct::C_CampaignType_Quarterly) {
								Trackmania_BrowserAPI_State.RequestsManager.WaitingForQuarterlyCampaigns = False;
							} else if (CampaignType == CampaignStruct::C_CampaignType_Monthly) {
								Trackmania_BrowserAPI_State.RequestsManager.WaitingForMonthlyCampaigns = False;
							}
						}
					} else if (Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMaps && Event.CustomEventType == MapStore::C_Event_FavoriteMapsUpdated) {
						Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMaps = False;
						Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].FavoriteMaps = GetFavoriteMapsData();
					}  else if (Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMapSupression && Event.CustomEventType == MapStore::C_Event_CurrentMapUpdated) {
						Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMapSupression = False;
						if (MapStore::GetCurrentMap().Status == MapStore::C_CurrentMapStatus_Error && MapStore::GetCurrentMap().ErrorCode == MapStore::C_ErrorCode_FailedRemoveFavorites) {
							ShowError(_This, C_RemoveFavoriteMap_Error);
						}
						Trackmania_BrowserAPI_State.FavoriteMapToRemove = K_MapBrowserData {};
						UpdateMapPreselection(_This, K_MapBrowserData {}, Trackmania_BrowserAPI_State.Controls.Frame_SelectedMap);
						Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMaps = True;
						Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
					} else if (
						Event.CustomEventType == CampaignStore::C_Event_ClubsCampaignListUpdated ||
						Event.CustomEventType == CampaignStore::C_Event_ClubsCampaignListError
					) {
						Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubCampaigns = False;
					} else if (
						Event.CustomEventType == ClubStore::C_Event_BrowserClubsUpdated ||
						Event.CustomEventType == ClubStore::C_Event_BrowserClubsError
					) {
						foreach	(BrowserType => Browser in Trackmania_BrowserAPI_State.Browsers) {
							Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent[CampaignStruct::C_CampaignType_Club] = Event.CustomEventType == ClubStore::C_Event_BrowserClubsUpdated;
						}
						Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubs = False;
					} else if (Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedCampaign &&
						(
							Event.CustomEventType == CampaignStore::C_Event_CampaignLoadComplete ||
							Event.CustomEventType == CampaignStore::C_Event_CampaignListUpdateError
						)
					) {
						Trackmania_BrowserAPI_State.RequestsManager.CampaignIsLoaded = Event.CustomEventType == CampaignStore::C_Event_CampaignLoadComplete;
						Trackmania_BrowserAPI_State.RequestsManager.CampaignLoadFail = !Trackmania_BrowserAPI_State.RequestsManager.CampaignIsLoaded;
					} else if (Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedMonthlyCampaign &&
						(
							Event.CustomEventType == CampaignStore::C_Event_MonthlyCampaignLoadComplete ||
							Event.CustomEventType == CampaignStore::C_Event_CampaignListUpdateError
						)
					) {
						Trackmania_BrowserAPI_State.RequestsManager.MonthlyCampaignIsLoaded = Event.CustomEventType == CampaignStore::C_Event_MonthlyCampaignLoadComplete;
						Trackmania_BrowserAPI_State.RequestsManager.MonthlyCampaignLoadFail = !Trackmania_BrowserAPI_State.RequestsManager.MonthlyCampaignIsLoaded;
					}
				}
			}
		}

		if (Trackmania_BrowserAPI_State.RequestsManager.CampaignIsLoaded) {
			Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedCampaign = False;
		}
		if (Trackmania_BrowserAPI_State.RequestsManager.MonthlyCampaignIsLoaded) {
			Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedMonthlyCampaign = False;
		}

		if (Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime > 0 && Now >= Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime) {
			declare ElementsPerPage = GetElementsPerPage(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
			declare Integer Length = ElementsPerPage * 3;

			LoadingSpinner::StartAnimation(_This, Trackmania_BrowserAPI_State.Controls.Frame_LoadingSpinner);
			UpdateBrowserContent(_This, Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll, 0);
			Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll.Visible = False;

			if (Trackmania_BrowserAPI_State.RequestsManager.WaitingForMonthlyCampaigns) {
				CampaignStore::Action_LoadMonthlyCampaignsPage(_This, Trackmania_BrowserAPI_State.RequestsManager.PageMonthly, ElementsPerPage);
			}
			if (Trackmania_BrowserAPI_State.RequestsManager.WaitingForQuarterlyCampaigns) {
				declare Integer Offset = ML::Max(0, (Trackmania_BrowserAPI_State.RequestsManager.PageQuarterly - 2)) * ElementsPerPage;
				CampaignStore::Action_LoadCampaigns(_This, CampaignStruct::C_CampaignType_Quarterly, Offset, Length);
			}
			if (Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubCampaigns) {
				declare Integer Offset = ML::Max(0, (Trackmania_BrowserAPI_State.RequestsManager.PageClubCampaigns - 2)) * ElementsPerPage;
				CampaignStore::Action_LoadClubsCampaigns(_This, Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenClubId, Offset, Length);
			}
			if (Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubs) {
				declare Integer Offset = ML::Max(0, (Trackmania_BrowserAPI_State.RequestsManager.PageClubs - 2)) * ElementsPerPage;
				ClubStore::Action_LoadClubsBrowser(_This, Offset, Length);
			}
			if (Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedCampaign && !Trackmania_BrowserAPI_State.RequestsManager.CampaignIsLoaded) {
				CampaignStore::Action_LoadCampaign(_This, Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign.Id, Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign.Club.Id);
			}
			if (Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedMonthlyCampaign && !Trackmania_BrowserAPI_State.RequestsManager.MonthlyCampaignIsLoaded) {
				CampaignStore::Action_LoadMonthlyCampaign(_This, Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenMonthlyCampaign.Id);
			}
			if (Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMaps) {
				// @TODO if LIVE API updated on mapType filter, adapt code here (see /trackmania-next/tmnext/-/issues/3735)
				declare Text MapTypeFilter = "";
				if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.FilterByMapType.count > 0) MapTypeFilter = Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.FilterByMapType[0];
				declare Integer Offset = ML::Max(0, (Trackmania_BrowserAPI_State.RequestsManager.PageFavorites - 2)) * ElementsPerPage;
				MapStore::Action_LoadFavoriteMapsPage(
					_This,
					Offset,
					Length,
					Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].IsSortingByNameElseByDate,
					Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].IsSortingFromAscendingOrder,
					MapTypeFilter,
					Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.ShowOnlyPlayableMaps,
					Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.ShowOnlyUserMaps
				);
			}
			Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = -1;
		}

		if (
			!Trackmania_BrowserAPI_State.RequestsManager.WaitingForMonthlyCampaigns &&
			!Trackmania_BrowserAPI_State.RequestsManager.WaitingForQuarterlyCampaigns &&
			!Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubCampaigns &&
			!Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubs &&
			!Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedCampaign &&
			!Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedMonthlyCampaign &&
			!Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMaps
		) {
			LoadingSpinner::StopAnimation(_This, Trackmania_BrowserAPI_State.Controls.Frame_LoadingSpinner);
			InitPathsTree(_This);
			if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign.Id != CampaignStruct::C_Campaign_NullId) {
				if (Trackmania_BrowserAPI_State.RequestsManager.CampaignLoadFail) {
					Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign = CampaignStruct::LibCampaignStruct_K_Campaign {
						Id = CampaignStruct::C_Campaign_NullId
					};
				} else {
					if (Trackmania_BrowserAPI_State.RequestsManager.GetCampaign) {
						declare Campaign = FilterCampaignMaps(
							Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings,
							CampaignStore::GetCampaignFromId(Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign.Id)
						);
						Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
							Type = C_EventType_Select,
							DataType = C_DataType_Campaign,
							Data = Campaign.tojson()
						});
						ResetBrowserCampaignsAndClubId(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
						Trackmania_BrowserAPI_State.RequestsManager.GetCampaign = False;
					} else {
						Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign = FilterCampaignMaps(
							Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings,
							CampaignStore::GetCampaignFromId(Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign.Id)
						);
					}
				}
			}
			if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenMonthlyCampaign.Id != CampaignStruct::C_Campaign_NullId) {
				if (Trackmania_BrowserAPI_State.RequestsManager.MonthlyCampaignLoadFail) {
					Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenMonthlyCampaign = CampaignStruct::LibCampaignStruct_K_MonthlyCampaign {
						Id = CampaignStruct::C_Campaign_NullId
					};
				} else {
					declare MonthlyCampaign = FilterCampaignMaps(
						Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings,
						CampaignStore::GetMonthlyCampaignFromId(Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenMonthlyCampaign.Id)
					);
					declare TmpDailyMaps = MonthlyCampaign.DailyMaps;
					foreach (DayIndex => DailyMap in TmpDailyMaps) {
						if (DailyMap.Map.Uid == "" || (DailyMap.Map.DownloadUrl == "" && DailyMap.Map.LocalFileName == "") || DailyMap.RelativeStart > 0) {
							MonthlyCampaign.DailyMaps.removekey(DayIndex);
						}
					}
					if (Trackmania_BrowserAPI_State.RequestsManager.GetCampaign) {
						Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
							Type = C_EventType_Select,
							DataType = C_DataType_MonthlyCampaign,
							Data = MonthlyCampaign.tojson()
						});
						ResetBrowserCampaignsAndClubId(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
						Trackmania_BrowserAPI_State.RequestsManager.GetCampaign = False;
					} else {
						Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenMonthlyCampaign = MonthlyCampaign; //< Be sure that `MonthlyCampaign` maps are filtered at this point
					}
				}
			}
			UpdateBrowserContent(_This, Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll, GetBrowserDataCount(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser));
		}
	}
}

Void Private_ProcessScrollEvents(CMlScript _This) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;

	foreach (Event in Scroll::GetPendingEvents(_This)) {
		if (Event.Scroll != Null && Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll != Null && Event.Scroll.ControlId == Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll.ControlId) {
			switch (Event.Type) {
				case Scroll::C_EventType_Update: {
					declare DataCount = GetBrowserDataCount(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
					declare CurrentBrowser = Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser];
					if (
						CurrentBrowser.ChosenCampaign.Id == CampaignStruct::C_Campaign_NullId &&
						CurrentBrowser.ChosenMonthlyCampaign.Id == CampaignStruct::C_Campaign_NullId &&
						CurrentBrowser.ChosenClubId == CampaignStruct::C_Campaign_NullId
					) {
						switch (CurrentBrowser.CurrentPath) {
							case C_Path_Quarterly: {
								if (Event.Page != Trackmania_BrowserAPI_State.RequestsManager.PageQuarterly) {
									Trackmania_BrowserAPI_State.RequestsManager.PageQuarterly = Event.Page;
									Trackmania_BrowserAPI_State.RequestsManager.WaitingForQuarterlyCampaigns = True;

									if (
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime < 0 ||
										Now >= Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime + C_DelayBetweenRequest
									) {
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
									} else {
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now + C_DelayBetweenRequest;
									}
								}
							}
							case C_Path_Monthly: {
								if (Event.Page != Trackmania_BrowserAPI_State.RequestsManager.PageMonthly) {
									Trackmania_BrowserAPI_State.RequestsManager.PageMonthly = Event.Page;
									Trackmania_BrowserAPI_State.RequestsManager.WaitingForMonthlyCampaigns = True;

									if (
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime < 0 ||
										Now >= Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime + C_DelayBetweenRequest
									) {
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
									} else {
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now + C_DelayBetweenRequest;
									}
								}
							}
							case C_Path_Clubs: {
								if (Event.Page != Trackmania_BrowserAPI_State.RequestsManager.PageClubs) {
									Trackmania_BrowserAPI_State.RequestsManager.PageClubs = Event.Page;
									Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubs = True;

									if (
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime < 0 ||
										Now >= Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime + C_DelayBetweenRequest
									) {
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
									} else {
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now + C_DelayBetweenRequest;
									}
								}
							}
							case C_Path_ChosenClub: {
								if (Event.Page != Trackmania_BrowserAPI_State.RequestsManager.PageClubCampaigns) {
									Trackmania_BrowserAPI_State.RequestsManager.PageClubCampaigns = Event.Page;
									Trackmania_BrowserAPI_State.RequestsManager.WaitingForClubCampaigns = True;

									if (
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime < 0 ||
										Now >= Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime + C_DelayBetweenRequest
									) {
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
									} else {
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now + C_DelayBetweenRequest;
									}
								}
							}
							case C_Path_Favorites: {
								if (Event.Page != Trackmania_BrowserAPI_State.RequestsManager.PageFavorites) {
									Trackmania_BrowserAPI_State.RequestsManager.PageFavorites = Event.Page;
									Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMaps = True;

									if (
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime < 0 ||
										Now >= Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime + C_DelayBetweenRequest
									) {
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
									} else {
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now + C_DelayBetweenRequest;
									}
								}
							}
						}
					}
					foreach (Key => Frame_Element in Event.ElementsVisible) {
						Frame_Element.Visible = DataCount > Key;
						if (Frame_Element.Visible) {
							UpdateBrowserElement(_This, Frame_Element, Key);
						}
					}
					foreach (Key => Frame_Element in Event.ElementsHidden) {
						Frame_Element.Visible = False;
					}
					Event.Scroll.Visible = Event.ElementsVisible.count != 0;

					// Hide other Label_NoElements
					declare Frame_BrowserMapScroll <=> (Trackmania_BrowserAPI_State.Controls.Frame_Global.Parent.GetFirstChild(C_Id_FrameBrowserMapScroll) as CMlFrame);
					declare Frame_BrowserTestMapScroll <=> (Trackmania_BrowserAPI_State.Controls.Frame_Global.Parent.GetFirstChild(C_Id_FrameBrowserSmallMapScroll) as CMlFrame);
					declare Frame_BrowserListScroll <=> (Trackmania_BrowserAPI_State.Controls.Frame_Global.Parent.GetFirstChild(C_Id_FrameBrowserListScroll) as CMlFrame);
					Frame_BrowserMapScroll.Parent.GetFirstChild(C_Id_LabelNoElements).Hide();
					Frame_BrowserTestMapScroll.Parent.GetFirstChild(C_Id_LabelNoElements).Hide();
					Frame_BrowserListScroll.Parent.GetFirstChild(C_Id_LabelNoElements).Hide();

					declare CMlLabel Label_NoElements = (Event.Scroll.Parent.GetFirstChild(C_Id_LabelNoElements) as CMlLabel);
					if (Label_NoElements != Null) Label_NoElements.Visible = !Event.Scroll.Visible && !LoadingSpinner::IsLoading(Trackmania_BrowserAPI_State.Controls.Frame_LoadingSpinner);
				}
				case Scroll::C_EventType_Focus: {
					if (Event.From != Null) FocusBrowserElement(_This, Event.From, False);
					if (Event.To != Null) {
						if (IsScrollFocused(_This) || Event.IsMouse) {
							FocusBrowserElement(_This, Event.To, True);
							Focus(_This, Null);

							if (
								Trackmania_BrowserAPI_State.Browsers.existskey(Trackmania_BrowserAPI_State.Type_CurrentBrowser) &&
								Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.ListMultiSelect
							) {
								if (GetDataType(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey) == C_DataType_Map) {
									declare K_MapBrowserData MapBrowserData = GetMapBrowserData(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey);
									UpdateMapPreselection(_This, MapBrowserData, Trackmania_BrowserAPI_State.Controls.Frame_SelectedMap);
								} else {
									UpdateMapPreselection(_This, K_MapBrowserData {}, Trackmania_BrowserAPI_State.Controls.Frame_SelectedMap);
								}
							}
						}
						if (!IsScrollFocused(_This)) {
							FocusScroll(_This, True);
						}
						if (!IsBrowserFocused(_This)) {
							FocusBrowser(_This, True);
						}
					}
				}
				case Scroll::C_EventType_ReachLimit: {
					if (Event.NavAction == CMlScriptEvent::EMenuNavAction::Up) {
						FocusBrowserElement(_This, Event.From, False);
						FocusScroll(_This, False);
						declare CMlControl ControlTarget = Navigation::GetControlTarget(Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll, Event.NavAction);
						if (ControlTarget != Null) {
							if (Navigation::IsLocked(ControlTarget)) {
								ControlTarget = Navigation::GetControlTarget(ControlTarget, Event.NavAction);
								if (Navigation::IsLocked(ControlTarget)) ControlTarget = Trackmania_BrowserAPI_State.Controls.Button_ParentPath;
							}
						}
						if (Trackmania_BrowserAPI_State.Controls.Frame_FocusHeader != Null && ControlTarget.HasClass(C_Class_Header)) {
							ControlTarget = Trackmania_BrowserAPI_State.Controls.Frame_FocusHeader;
						}

						Focus(_This, ControlTarget);
						MenuSounds::PlayFocusSound(_This, ControlTarget);
					} else {
						declare CMlControl Button_EditorSettings = Trackmania_BrowserAPI_State.Controls.Frame_SelectedMap.GetFirstChild(C_Id_ButtonEditorSettings);
						declare CMlControl Button_SelectMap = Trackmania_BrowserAPI_State.Controls.Frame_SelectedMap.GetFirstChild(C_Id_ButtonSelectMap);
						declare Frame_ButtonCreateMap <=> (Trackmania_BrowserAPI_State.Controls.Frame_Global.GetFirstChild(C_Id_ButtonCreateMap) as CMlFrame);
						if (Event.NavAction == CMlScriptEvent::EMenuNavAction::Right && !Navigation::IsLocked(Button_EditorSettings)) {
							FocusBrowserElement(_This, Event.From, False);
							FocusScroll(_This, False);
							Focus(_This, Button_EditorSettings);
						} else if (
							Event.NavAction == CMlScriptEvent::EMenuNavAction::Down &&
							(
								!Navigation::IsLocked(Trackmania_BrowserAPI_State.Controls.Button_ListMultiSelectValid) ||
								!Navigation::IsLocked(Button_SelectMap) ||
								!Navigation::IsLocked(Frame_ButtonCreateMap)
							)
						) {
							FocusBrowserElement(_This, Event.From, False);
							FocusScroll(_This, False);
							declare ControlTarget = Navigation::GetControlTarget(Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll, Event.NavAction);
							if (!Navigation::IsLocked(Trackmania_BrowserAPI_State.Controls.Button_ListMultiSelectValid)) {
								ControlTarget = Trackmania_BrowserAPI_State.Controls.Button_ListMultiSelectValid;
							} else if (!Navigation::IsLocked(Button_SelectMap)){
								ControlTarget = Button_SelectMap;
							} else if (!Navigation::IsLocked(Frame_ButtonCreateMap)) {
								ControlTarget = Frame_ButtonCreateMap;
							} else {
								Trackmania_BrowserAPI_State.PendingEvents.add(K_Event {
									Type = C_EventType_ReachLimit,
									From = Event.From,
									MenuNavAction = Event.NavAction
								});
							}

							if (Trackmania_BrowserAPI_State.Controls.Frame_FocusFooter != Null && !Navigation::IsLocked(Trackmania_BrowserAPI_State.Controls.Frame_FocusFooter) && ControlTarget.HasClass(C_Class_Footer)) {
								ControlTarget = Trackmania_BrowserAPI_State.Controls.Frame_FocusFooter;
							}
							if (!Navigation::IsLocked(ControlTarget)) Focus(_This, ControlTarget);
						} else {
							Trackmania_BrowserAPI_State.PendingEvents.add(K_Event {
								Type = C_EventType_ReachLimit,
								From = Event.From,
								MenuNavAction = Event.NavAction
							});
						}
					}
				}
				case Scroll::C_EventType_Select: {
					ShowError(_This, "");
					switch (GetDataType(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey)) {
						case C_DataType_Campaign: {
							declare Campaign = GetCampaign(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey);
							declare HasPermission = True;
							declare PermissionMessage = "";
							declare LiveQuarterlyCampaign = CampaignStore::GetLiveQuarterlyCampaign();

							if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.IsBrowsingForEdition) {
								HasPermission = UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_CreateAndUploadMap);
								if (!HasPermission) PermissionMessage = RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_OpenAdvancedMapEditor);
							} else if (Campaign.Id == LiveQuarterlyCampaign.Id) {
								HasPermission = UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_PlayCurrentOfficialQuarterlyCampaign);
								if (!HasPermission) PermissionMessage = RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_PlayCurrentOfficialQuarterlyCampaign);
							} else {
								HasPermission = UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_PlayPastOfficialQuarterlyCampaign);
								if (!HasPermission) PermissionMessage = RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_PlayPastOfficialQuarterlyCampaign);
							}
							if (!HasPermission) {
								ClubEdition::GoToPage(_This, Router::GetCurrentRoutePath(_This), PermissionMessage);
							} else {
								if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Campaign)) {
									if (Campaign.IsLoaded) {
										Campaign = CampaignStore::GetCampaignFromId(Campaign.Id);
										Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
											Type = C_EventType_Select,
											DataType = C_DataType_Campaign,
											Data = Campaign.tojson()
										});
										ResetBrowserCampaignsAndClubId(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
									} else {
										Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign = FilterCampaignMaps(
											Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings,
											Campaign
										);
										Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedCampaign = True;
										Trackmania_BrowserAPI_State.RequestsManager.GetCampaign = True;
										Trackmania_BrowserAPI_State.RequestsManager.CampaignIsLoaded = False;
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
									}
								} else if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Map)) {
									Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenCampaign = FilterCampaignMaps(
										Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings,
										Campaign
									);
									Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedCampaign = True;
									Trackmania_BrowserAPI_State.RequestsManager.CampaignIsLoaded = Campaign.IsLoaded;
									if (
										!Campaign.IsLoaded && (
											Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].CurrentPath == C_Path_ChosenClub ||
											Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].CurrentPath == C_Path_Quarterly
										)
									) {
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
									}
								}
							}
						}
						case C_DataType_MonthlyCampaign: {
							declare MonthlyCampaign = GetMonthlyCampaign(Event.ToKey);
							declare HasPermission = True;
							declare PermissionMessage = "";
							declare LiveMonthlyCampaign = CampaignStore::GetLiveMonthlyCampaign();

							if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.IsBrowsingForEdition) {
								HasPermission = UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_CreateAndUploadMap);
								if (!HasPermission) PermissionMessage = RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_OpenAdvancedMapEditor);
							} else if (MonthlyCampaign.Id == LiveMonthlyCampaign.Id) {
								HasPermission = UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_PlayCurrentOfficialMonthlyCampaign);
								if (!HasPermission) PermissionMessage = RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_PlayCurrentOfficialMonthlyCampaign);
							} else {
								HasPermission = UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_PlayPastOfficialMonthlyCampaign);
								if (!HasPermission) PermissionMessage = RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_PlayPastOfficialMonthlyCampaign);
							}
							if (!HasPermission) {
								ClubEdition::GoToPage(_This, Router::GetCurrentRoutePath(_This), PermissionMessage);
							} else {
								if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_MonthlyCampaign)) {
									if (MonthlyCampaign.IsLoaded) {
										MonthlyCampaign = CampaignStore::GetMonthlyCampaignFromId(MonthlyCampaign.Id);
										Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
											Type = C_EventType_Select,
											DataType = C_DataType_MonthlyCampaign,
											Data = MonthlyCampaign.tojson()
										});
										ResetBrowserCampaignsAndClubId(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser);
									} else {
										Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenMonthlyCampaign = FilterCampaignMaps(
											Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings,
											MonthlyCampaign
										);
										Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedMonthlyCampaign = True;
										Trackmania_BrowserAPI_State.RequestsManager.GetCampaign = True;
										Trackmania_BrowserAPI_State.RequestsManager.MonthlyCampaignIsLoaded = False;
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
									}
								} else if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].DataTypes.exists(C_DataType_Map)) {
									Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenMonthlyCampaign = FilterCampaignMaps(
										Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings,
										MonthlyCampaign
									);
									Trackmania_BrowserAPI_State.RequestsManager.WaitingForLoadedMonthlyCampaign = True;
									Trackmania_BrowserAPI_State.RequestsManager.MonthlyCampaignIsLoaded = MonthlyCampaign.IsLoaded;
									if (
										!MonthlyCampaign.IsLoaded &&
										Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].CurrentPath == C_Path_Monthly
									) {
										Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = Now;
									}
								}
							}
						}
						case C_DataType_Map: {
							declare K_MapBrowserData MapBrowserData = GetMapBrowserData(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey);
							if (MapBrowserData.SubFolder != "") {
								SetCurrentPath(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, MapBrowserData.SubFolder);
							} else {
								declare HasPermission = True;
								declare PermissionMessage = "";
								if (!MapBrowserData.MapInfo.CreatedWithSimpleEditor && Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.IsBrowsingForEdition) {
									HasPermission = UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_OpenAdvancedMapEditor);
									if (!HasPermission) PermissionMessage = RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_OpenAdvancedMapEditor);
								}
								if (!HasPermission) {
									ClubEdition::GoToPage(_This, Router::GetCurrentRoutePath(_This), PermissionMessage);
								} else if (Trackmania_BrowserAPI_State.Browsers.existskey(Trackmania_BrowserAPI_State.Type_CurrentBrowser) && Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.ListMultiSelect) {
									ToggleMapSelection(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, MapBrowserData.MapInfo, Trackmania_BrowserAPI_State.Controls.Button_ListMultiSelectValid);
									UpdateBrowserElement(_This, Event.To, Event.ToKey);
									FocusBrowserElement(_This, Event.To, True);
								} else {
									Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
										Type = C_EventType_Select,
										DataType = C_DataType_Map,
										Data = MapBrowserData.MapInfo.tojson()
									});
									UpdateMapPreselection(_This, MapBrowserData, Trackmania_BrowserAPI_State.Controls.Frame_SelectedMap);
								}
							}
						}
						case C_DataType_Replay: {
							declare K_ReplayBrowserData ReplayBrowserData = GetReplayBrowserData(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey);
							if (ReplayBrowserData.SubFolder != "") {
								SetCurrentPath(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, ReplayBrowserData.SubFolder);
								UpdateBrowserContent(_This, Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll, GetBrowserDataCount(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser));
							} else {
								declare Boolean BrowserIsValid = Trackmania_BrowserAPI_State.Browsers.existskey(Trackmania_BrowserAPI_State.Type_CurrentBrowser);
								if (
									BrowserIsValid &&
									Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.IsPlayAgainstReplay &&
									!UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_PlayEntireOfficialQuarterlyCampaign) &&
									RBAC::TrackRequiresEntireCampaignAccess(CampaignStore::GetLiveQuarterlyCampaign(), ReplayBrowserData.ReplayInfo.MapUid)
								) {
									ClubEdition::GoToPage(_This, Router::GetCurrentRoutePath(_This), RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_PlayEntireOfficialQuarterlyCampaign));
								} else if (BrowserIsValid && Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.ListMultiSelect) {
									ToggleReplaySelection(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, ReplayBrowserData.ReplayInfo, Trackmania_BrowserAPI_State.Controls.Button_ListMultiSelectValid);
									UpdateBrowserElement(_This, Event.To, Event.ToKey);
									FocusBrowserElement(_This, Event.To, True);
								} else {
									Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
										Type = C_EventType_Select,
										DataType = C_DataType_Replay,
										Data = ReplayBrowserData.ReplayInfo.tojson()
									});
								}
							}
						}
						case C_DataType_Image: {
							declare FileBrowserData = GetFileBrowserData(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey);
							if (FileBrowserData.SubFolder != "") {
								SetCurrentPath(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, FileBrowserData.SubFolder);
							} else {
								Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
									Type = C_EventType_Select,
									DataType = C_DataType_Image,
									Data = FileBrowserData.File
								});
							}
						}
						case C_DataType_Skin: {
							DestroyScene(_This);
							declare FileBrowserData = GetFileBrowserData(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey);
							if (FileBrowserData.SubFolder != "") {
								SetCurrentPath(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, FileBrowserData.SubFolder);
							} else {
								Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
									Type = C_EventType_Select,
									DataType = C_DataType_Skin,
									Data = FileBrowserData.File
								});
							}
						}
						case C_DataType_Script: {
							declare FileBrowserData = GetFileBrowserData(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey);
							if (FileBrowserData.SubFolder != "") {
								SetCurrentPath(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, FileBrowserData.SubFolder);
							} else {
								Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
									Type = C_EventType_Select,
									DataType = C_DataType_Script,
									Data = FileBrowserData.File
								});
							}
						}
						case C_DataType_MatchSettings: {
							declare FileBrowserData = GetFileBrowserData(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey);
							if (FileBrowserData.SubFolder != "") {
								SetCurrentPath(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, FileBrowserData.SubFolder);
							} else {
								Trackmania_BrowserAPI_State.PendingEvents.add(K_Event{
									Type = C_EventType_Select,
									DataType = C_DataType_MatchSettings,
									Data = FileBrowserData.File
								});
							}
						}
						case C_DataType_Club: {
							if (!UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_JoinClub)) {
								declare PermissionMessage = "";
								if (Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.IsBrowsingForEdition) {
									PermissionMessage = RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_EditClubMap);
								} else {
									PermissionMessage = RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_PlayPrivateActivity);
								}
								ClubEdition::GoToPage(_This, Router::GetCurrentRoutePath(_This), PermissionMessage);
							} else {
								Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].ChosenClubId = GetClub(Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey).Id;
								SetCurrentPath(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, C_Path_ChosenClub);
							}
						}
						case C_DataType_ItemCollection: {
							declare FileBrowserData = GetFileBrowserData(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey);
							if (FileBrowserData.SubFolder != "") {
								SetCurrentPath(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, FileBrowserData.SubFolder);
							} else {
								Trackmania_BrowserAPI_State.PendingEvents.add(K_Event {
									Type = C_EventType_Select,
									DataType = C_DataType_ItemCollection,
									Data = FileBrowserData.File
								});
							}
						}
						case C_DataType_Category: {
							declare CurrentPath = Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].CurrentPath;
							if (Trackmania_BrowserAPI_State.Paths_Tree.existskey(CurrentPath) && Trackmania_BrowserAPI_State.Paths_Tree[CurrentPath].existskey(Event.ToKey)) {
								SetCurrentPath(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Trackmania_BrowserAPI_State.Paths_Tree[CurrentPath][Event.ToKey]);
							}
							UpdateBrowserContent(_This, Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll, GetBrowserDataCount(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser));
						}
					}
				}
				case Scroll::C_EventType_Cancel: {
					if (IsScrollFocused(_This)) Cancel(_This, True);
				}
				case Scroll::C_EventType_NavAction: {
				if (
					Event.NavAction == CMlScriptEvent::EMenuNavAction::Action1 &&
					Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].Settings.ShowFavoriteUserMaps &&
					Trackmania_BrowserAPI_State.Browsers[Trackmania_BrowserAPI_State.Type_CurrentBrowser].CurrentPath == C_Path_Favorites
					) {
						if (GetDataType(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey) == C_DataType_Map) {
							Trackmania_BrowserAPI_State.FavoriteMapToRemove = GetMapBrowserData(_This, Trackmania_BrowserAPI_State.Type_CurrentBrowser, Event.ToKey);
							RemoveMapFromFavorites(_This, Trackmania_BrowserAPI_State.FavoriteMapToRemove, Trackmania_BrowserAPI_State.Controls.Frame_MultiChoice);
						}
					}
				}
			}
		}
	}
}

K_Event[] GetPendingEvents(CMlScript _This) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	return Trackmania_BrowserAPI_State.PendingEvents;
}

Void MainInit(CMlScript _This) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	Trackmania_BrowserAPI_State = C_State_Null;

	Trackmania_BrowserAPI_State.Controls.Frame_Global <=> (_This.Page.GetFirstChild(C_Id_FrameGlobal) as CMlFrame);
	Trackmania_BrowserAPI_State.RequestsManager.DisplayLiveContent = [
		CampaignStruct::C_CampaignType_Quarterly => False,
		CampaignStruct::C_CampaignType_Monthly => False,
		CampaignStruct::C_CampaignType_Club => False
	];
	InitPathsTree(_This);
	Trackmania_BrowserAPI_State.RequestsManager.NextPageLoadTime = -1;
}

Void MainStart(CMlScript _This) {
	declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
	if (Trackmania_BrowserAPI_State.Controls.Frame_Global != Null) {
		Trackmania_BrowserAPI_State.Controls.Frame_SelectedMap <=> (Trackmania_BrowserAPI_State.Controls.Frame_Global.GetFirstChild(C_Id_FrameSelectedMap) as CMlFrame);
		declare CMlFrame Frame_ListButtons <=> (Trackmania_BrowserAPI_State.Controls.Frame_Global.GetFirstChild(C_Id_FrameListButtons) as CMlFrame);
		Trackmania_BrowserAPI_State.Controls.Button_ListMultiSelectValid <=> (Frame_ListButtons.GetFirstChild(C_Id_ButtonListMultiSelectValid) as CMlFrame);
		Trackmania_BrowserAPI_State.Controls.Button_ParentPath <=> (Trackmania_BrowserAPI_State.Controls.Frame_Global.GetFirstChild(C_Id_FrameParentPath) as CMlFrame);
		Trackmania_BrowserAPI_State.Controls.Frame_LoadingSpinner <=> (Trackmania_BrowserAPI_State.Controls.Frame_Global.GetFirstChild(C_Id_FrameLoadingSpinner) as CMlFrame);
		Trackmania_BrowserAPI_State.Controls.Frame_MultiChoice = (Trackmania_BrowserAPI_State.Controls.Frame_Global.GetFirstChild(C_Id_FrameMultichoice) as CMlFrame);

		Trackmania_BrowserAPI_State.ManialinkProps.ElementsPerPage = TL::ToInteger(Trackmania_BrowserAPI_State.Controls.Frame_Global.DataAttributeGet("elements-per-page"));
		Trackmania_BrowserAPI_State.ManialinkProps.SmallMapsPerPage = TL::ToInteger(Trackmania_BrowserAPI_State.Controls.Frame_Global.DataAttributeGet("small-maps-per-page"));
		Trackmania_BrowserAPI_State.ManialinkProps.ListsPerPage = TL::ToInteger(Trackmania_BrowserAPI_State.Controls.Frame_Global.DataAttributeGet("lists-per-page"));
		Trackmania_BrowserAPI_State.ManialinkProps.SizeX_ListMargin = TL::ToReal(Trackmania_BrowserAPI_State.Controls.Frame_Global.DataAttributeGet("size-x-list-margin"));
		Trackmania_BrowserAPI_State.ManialinkProps.Size_BrowserLists = Tools::TextToVec2(Trackmania_BrowserAPI_State.Controls.Frame_Global.DataAttributeGet("size-browser-lists"));
	}
}

Void MainLoop(CMlScript _This) {
	if (_This.PageIsVisible) {
		declare K_State Trackmania_BrowserAPI_State for _This.Page = C_State_Null;
		Trackmania_BrowserAPI_State.PendingEvents = [];
		foreach	(BrowserType => Browser in Trackmania_BrowserAPI_State.Browsers) {
			if (Browser.DataTypes.exists(C_DataType_Map)) {
				declare CTaskResult_MapList TaskResultList;
				if (Browser.TaskId_GetMaps != NullId) {
					if (DataFileMgr.TaskResults.existskey(Browser.TaskId_GetMaps)) {
						TaskResultList = (DataFileMgr.TaskResults[Browser.TaskId_GetMaps] as CTaskResult_MapList);
					} else {
						Trackmania_BrowserAPI_State.Browsers[BrowserType].MapTaskListState = C_TaskListState_HasFailed;
						Trackmania_BrowserAPI_State.Browsers[BrowserType].TaskId_GetMaps = NullId;
					}
				}
				if (TaskResultList != Null) {
					if (TaskResultList.HasSucceeded) {
						declare K_MapBrowserData[] NewMapBrowserData;
						foreach (SubFolder in TaskResultList.SubFolders) {
							NewMapBrowserData.add(K_MapBrowserData{
								SubFolder = SubFolder,
								Path = TaskResultList.Path,
								ParentPath = TaskResultList.ParentPath
							});
						}
						foreach (MapInfo in TaskResultList.MapInfos) {
							// @TODO check if the folders are empty and do not display them ?
							// It would mean checking every folder... Maybe ask Falguière a boolean to filter the maps ?
							if (FilterMap(Browser.Settings, MapInfo.AuthorLogin, MapInfo.IsPlayable, MapInfo.CollectionName, MapInfo.MapType)) {
								NewMapBrowserData.add(K_MapBrowserData{
									MapInfo = CampaignStruct::LibCampaignStruct_K_Map {
										Uid = MapInfo.MapUid,
										Name = MapInfo.Name,
										Type = MapInfo.MapType,
										Style = MapInfo.MapStyle,
										AuthorLogin = MapInfo.AuthorLogin,
										AuthorDisplayName = MapInfo.AuthorNickName,
										AuthorTime = MapInfo.TMObjective_AuthorTime,
										GoldTime = MapInfo.TMObjective_GoldTime,
										SilverTime = MapInfo.TMObjective_SilverTime,
										BronzeTime = MapInfo.TMObjective_BronzeTime,
										ThumbnailUrl = "file://Thumbnails/MapUid/"^MapInfo.MapUid,
										IsPlayable = MapInfo.IsPlayable,
										CollectionName = MapInfo.CollectionName,
										LocalFileName = MapInfo.FileName,
										CreatedWithSimpleEditor = MapInfo.CreatedWithSimpleEditor,
										CreatedWithGamepadEditor = MapInfo.CreatedWithGamepadEditor
									},
									Path = TaskResultList.Path,
									ParentPath = TaskResultList.ParentPath
								});
							}
						}
						Trackmania_BrowserAPI_State.Browsers[BrowserType].Maps = [];
						UpdateLocalMapsData(_This, BrowserType, NewMapBrowserData);
						Trackmania_BrowserAPI_State.Browsers[BrowserType].MapTaskListState = C_TaskListState_HasSucceeded;
					} else if (TaskResultList.HasFailed) {
						Trackmania_BrowserAPI_State.Browsers[BrowserType].MapTaskListState = C_TaskListState_HasFailed;
					} else if (TaskResultList.IsCanceled) {
						Trackmania_BrowserAPI_State.Browsers[BrowserType].MapTaskListState = C_TaskListState_IsCanceled;
					}
					if (!TaskResultList.IsProcessing) {
						DataFileMgr.TaskResult_Release(TaskResultList.Id);
						Trackmania_BrowserAPI_State.Browsers[BrowserType].TaskId_GetMaps = NullId;
					} else {
						Trackmania_BrowserAPI_State.Browsers[BrowserType].MapTaskListState = C_TaskListState_IsProcessing;
					}
				}
			}
			if (Browser.DataTypes.exists(C_DataType_Replay)) {
				declare CTaskResult_ReplayList TaskResultList;
				if (Browser.TaskId_GetReplays != NullId) {
					if (DataFileMgr.TaskResults.existskey(Browser.TaskId_GetReplays)) {
						TaskResultList = (DataFileMgr.TaskResults[Browser.TaskId_GetReplays] as CTaskResult_ReplayList);
					} else {
						Trackmania_BrowserAPI_State.Browsers[BrowserType].ReplayTaskListState = C_TaskListState_HasFailed;
						Trackmania_BrowserAPI_State.Browsers[BrowserType].TaskId_GetReplays = NullId;
					}
				}
				if (TaskResultList != Null) {
					if (TaskResultList.HasSucceeded) {
						declare K_ReplayBrowserData[] NewReplays;
						foreach (SubFolder in TaskResultList.SubFolders) {
							NewReplays.add(K_ReplayBrowserData {
								SubFolder = SubFolder,
								ReplayInfo = K_ReplayInfo {}
							});
						}
						foreach (ReplayInfo in TaskResultList.ReplayInfos) {
							NewReplays.add(K_ReplayBrowserData {
								SubFolder = "",
								ReplayInfo = K_ReplayInfo {
									Id = ReplayInfo.MapUid^"\\"^ReplayInfo.FileName,
									MapUid = ReplayInfo.MapUid,
									Name = ReplayInfo.Name,
									Path = ReplayInfo.Path,
									FileName = ReplayInfo.FileName
								}
							});
						}

						Trackmania_BrowserAPI_State.Browsers[BrowserType].Replays = [];
						UpdateLocalReplaysData(_This, BrowserType, NewReplays);
						Trackmania_BrowserAPI_State.Browsers[BrowserType].ReplayTaskListState = C_TaskListState_HasSucceeded;
					} else if (TaskResultList.HasFailed) {
						Trackmania_BrowserAPI_State.Browsers[BrowserType].ReplayTaskListState = C_TaskListState_HasFailed;
					} else if (TaskResultList.IsCanceled) {
						Trackmania_BrowserAPI_State.Browsers[BrowserType].ReplayTaskListState = C_TaskListState_IsCanceled;
					}
					if (!TaskResultList.IsProcessing) {
						DataFileMgr.TaskResult_Release(TaskResultList.Id);
						Trackmania_BrowserAPI_State.Browsers[BrowserType].TaskId_GetReplays = NullId;
					} else {
						Trackmania_BrowserAPI_State.Browsers[BrowserType].ReplayTaskListState = C_TaskListState_IsProcessing;
					}
				}
			}
			foreach (DataType => File in Browser.Files) {
				if (!Browser.DataTypes.exists(DataType)) continue;
				declare CTaskResult_FileList TaskResultList;
				if (File.TaskId_GetFiles != NullId) {
					if (DataFileMgr.TaskResults.existskey(File.TaskId_GetFiles)) {
						TaskResultList = (DataFileMgr.TaskResults[File.TaskId_GetFiles] as CTaskResult_FileList);
					} else {
						Trackmania_BrowserAPI_State.Browsers[BrowserType].Files[DataType].TaskListState = C_TaskListState_HasFailed;
						Trackmania_BrowserAPI_State.Browsers[BrowserType].Files[DataType].TaskId_GetFiles = NullId;
					}
				}
				if (TaskResultList != Null) {
					if (TaskResultList.HasSucceeded) {
						declare K_FileBrowserData[] NewFileBrowserData;
						foreach (SubFolder in TaskResultList.SubFolders) {
							NewFileBrowserData.add(K_FileBrowserData{
								SubFolder = SubFolder,
								Path = TaskResultList.Path,
								ParentPath = TaskResultList.ParentPath
							});
						}
						foreach (File in TaskResultList.Files) {
							declare Boolean AddFile = True;
							if (File != "") {
								// The club themes filter is independent of the file extensions filter
								if (Browser.Settings.FilterClubThemeByMediaType.count > 0 && ClubStruct::IsClubThemeUrl(C_BasePath^C_BasePath_Media^File)) {
									AddFile = Browser.Settings.FilterClubThemeByMediaType.exists(ClubStruct::GetClubThemeType(C_BasePath^C_BasePath_Media^File));
								} else if (Browser.Settings.FilterByExtension.count > 0) {
									declare Text[] TmpSplit = TL::Split(".", TL::ToLowerCase(File));
									declare Text Extension = TmpSplit[TmpSplit.count-1];
									AddFile = Extension != "" && Browser.Settings.FilterByExtension.exists(Extension);
								}

							}
							if (AddFile) {
								NewFileBrowserData.add(K_FileBrowserData{
									File = File,
									Path = TaskResultList.Path,
									ParentPath = TaskResultList.ParentPath
								});
							}
						}
						Trackmania_BrowserAPI_State.Browsers[BrowserType].Files[DataType].Datas = [];
						SetLocalData(_This, BrowserType, DataType, NewFileBrowserData);
						Trackmania_BrowserAPI_State.Browsers[BrowserType].Files[DataType].TaskListState = C_TaskListState_HasSucceeded;
					} else if (TaskResultList.HasFailed) {
						Trackmania_BrowserAPI_State.Browsers[BrowserType].Files[DataType].TaskListState = C_TaskListState_HasFailed;
					} else if (TaskResultList.IsCanceled) {
						Trackmania_BrowserAPI_State.Browsers[BrowserType].Files[DataType].TaskListState = C_TaskListState_IsCanceled;
					}
					if (!TaskResultList.IsProcessing) {
						DataFileMgr.TaskResult_Release(TaskResultList.Id);
						Trackmania_BrowserAPI_State.Browsers[BrowserType].Files[DataType].TaskId_GetFiles = NullId;
					} else {
						Trackmania_BrowserAPI_State.Browsers[BrowserType].Files[DataType].TaskListState = C_TaskListState_IsProcessing;
					}
				}
			}
		}

		Private_UpdateRequestManager(_This);

		foreach (Event in PopUpBackground::GetPendingEvents(_This)) {
			if (Event.PopUpBackground != Null && Event.PopUpBackground.ControlId == C_Id_FramePopUpBackground) {
				switch (Event.Type) {
					case PopUpBackground::C_EventType_Close: LeaveBrowser(_This);
				}
			}
		}

		foreach (Event in Navigation::GetPendingEvents(_This)) {
			switch (Event.Type) {
				case Navigation::C_EventType_NavigateInput: {
					if (Navigation::IsEventFromGroup(_This, Event, C_NavGroupName) && IsBrowserFocused(_This) && !IsScrollFocused(_This) && !Trackmania_BrowserAPI_State.Controls.Frame_MultiChoice.Visible) {
						if (Event.Input == CMlScriptEvent::EMenuNavAction::Select) {
							Select(_This, Event.To);
						} else if (Event.Input == CMlScriptEvent::EMenuNavAction::Cancel) {
							Cancel(_This, True);
						} else {
							if (Event.To != Null && Event.From != Null) {
								if (Trackmania_BrowserAPI_State.Controls.Frame_FocusHeader != Null && Event.To.HasClass(C_Class_Header) && !Event.From.HasClass(C_Class_Header)) {
									Focus(_This, Trackmania_BrowserAPI_State.Controls.Frame_FocusHeader);
								} else if (Trackmania_BrowserAPI_State.Controls.Frame_FocusBody != Null && Event.To.HasClass(C_Class_Body) && !Event.From.HasClass(C_Class_Body)) {
									Focus(_This, Trackmania_BrowserAPI_State.Controls.Frame_FocusBody);
								} else if (Trackmania_BrowserAPI_State.Controls.Frame_FocusFooter != Null && Event.To.HasClass(C_Class_Footer) && !Event.From.HasClass(C_Class_Footer)) {
									Focus(_This, Trackmania_BrowserAPI_State.Controls.Frame_FocusFooter);
								} else {
									Focus(_This, Event.To);
								}

								if (Event.From.ControlId == Event.To.ControlId) {
									Trackmania_BrowserAPI_State.PendingEvents.add(K_Event {
										Type = C_EventType_ReachLimit,
										From = Event.From,
										MenuNavAction = Event.Input
									});
								}
							} else {
								Focus(_This, Event.To);
							}
						}
					}
				}
				case Navigation::C_EventType_NavigateMouse: {
					if (Navigation::IsEventFromGroup(_This, Event, C_NavGroupName)) {
						if (Event.Mouse == CMlScriptEvent::Type::MouseClick) {
							Select(_This, Event.To);
						} else if (Event.Mouse == CMlScriptEvent::Type::MouseOver) {
							FocusScroll(_This, False);
							Focus(_This, Event.To);
						}
					}
				}
			}
		}

		/* Pop up multi choice events */
		foreach (Event in PopUpMultiChoice::GetPendingEvents(_This)) {
			if (Event.PopUpMultiChoice != Null && Event.PopUpMultiChoice.Id == Trackmania_BrowserAPI_State.Controls.Frame_MultiChoice.Id) {
				switch (Event.Type) {
					case PopUpMultiChoice::C_EventType_Select: {
						if (Event.SelectedIndex == C_FavoriteMapPopUp_ChoiceIndex_Yes) {
							MapStore::Action_RemoveMapFromFavorites(_This, Trackmania_BrowserAPI_State.FavoriteMapToRemove.MapInfo.Uid);
							Trackmania_BrowserAPI_State.RequestsManager.WaitingForFavoriteMapSupression = True;
						}
					}
					case PopUpMultiChoice::C_EventType_Display: {
						Navigation::EnableGroup(_This, C_NavGroupName, !Trackmania_BrowserAPI_State.Controls.Frame_MultiChoice.Visible);
						if (Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll != Null) Scroll::LockNavigation(Trackmania_BrowserAPI_State.Controls.Frame_CurrentScroll, Trackmania_BrowserAPI_State.Controls.Frame_MultiChoice.Visible);
					}
				}
			}
		}

		Private_ProcessScrollEvents(_This);
	}
}
