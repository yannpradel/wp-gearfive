/**
 *  UI module: InfoPanels_2x3 server side
 */
#Const Version    "1.0.0"
#Const ScriptName "Libs/Nadeo/Trackmania/Modes/ChampionTMGL/UIModules/InfoPanels_2x3_Server.Script.txt"

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// Constants
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
#Const C_Id "UIModule_ChampionTMGL_InfoPanels_2x3"

#Const C_RankingMode_BestRace 0
#Const C_RankingMode_CurrentRace 1
#Const C_RankingMode_MaxPoints 2

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// Functions
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// Public
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Update the race state
Void SetWarmUp(Boolean _IsWarmUp) {
	declare netwrite Boolean Net_ChampionTMGL_InfoPanels_2x3_IsWarmUp for Teams[0];
	Net_ChampionTMGL_InfoPanels_2x3_IsWarmUp = _IsWarmUp;
}

/// Select how the players are ranked
Void SetRankingMode(Integer _RankingMode) {
	declare netwrite Integer Net_ChampionTMGL_InfoPanels_2x3_RankingMode for Teams[0];
	Net_ChampionTMGL_InfoPanels_2x3_RankingMode = _RankingMode;
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/** Show or hide the UI
 *
 *	@param	_Player										The player to update
 *	@param	_Display									True to show
 *																		False to hide
 */
Void SetVisibility(CPlayer _Player, Boolean _Display) {
	if (_Player == Null) return;

	declare netwrite Boolean Net_ChampionTMGL_InfoPanels_2x3_IsVisible for _Player;
	Net_ChampionTMGL_InfoPanels_2x3_IsVisible = _Display;
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/** Initialize a player
 *
 *	@param	_Player										The player to initialize
 */
Void InitializePlayer(CPlayer _Player) {
	if (_Player == Null) return;

	SetVisibility(_Player, False);
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Unload the library
Void Unload() {
	SetWarmUp(False);
	SetRankingMode(C_RankingMode_BestRace);
	foreach (Player in AllPlayers) {
		InitializePlayer(Player);
	}
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Load the library
Void Load() {
	Unload();
}
