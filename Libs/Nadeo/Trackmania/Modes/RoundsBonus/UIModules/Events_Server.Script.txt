#Const Version		"1.0.0"
#Const ScriptName	"Libs/Nadeo/Trackmania/Modes/RoundsBonus/UIModules/Events_Server.Script.txt"

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// Includes
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
#Include "Libs/Nadeo/CMGame/Modes/UIModules_Common.Script.txt" as UIModules_Common
#Include "Libs/Nadeo/Trackmania/Modes/RoundsBonus/UIModules/Events_Common.Script.txt" as UIModules_Events_Common
#Include "Libs/Nadeo/Trackmania/Modes/RoundsBonus/Data.Script.txt" as Data

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// Functions
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// Public
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Get the id of the UI module
Text GetId() {
	return UIModules_Events_Common::C_Id;
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Get the configuration of the UI module
UIModules_Common::K_ModuleConfig GetConfig() {
	return UIModules_Events_Common::C_UIModuleConfig;
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Update the event situation
Void UpdateEvent(Integer _ActiveEventType, Integer _ActiveEventEndTime, Integer _NextEventType, Boolean _NextEventAnnounced, Integer _NextEventStartTime) {
	declare netwrite Integer Net_RoundsBonus_Events_ActiveEventType for Teams[0];
	declare netwrite Integer Net_RoundsBonus_Events_ActiveEventEndTime for Teams[0];
	declare netwrite Integer Net_RoundsBonus_Events_NextEventType for Teams[0];
	declare netwrite Boolean Net_RoundsBonus_Events_NextEventAnnounced for Teams[0];
	declare netwrite Integer Net_RoundsBonus_Events_NextEventStartTime for Teams[0];
	Net_RoundsBonus_Events_ActiveEventType = _ActiveEventType;
	Net_RoundsBonus_Events_ActiveEventEndTime = _ActiveEventEndTime;
	Net_RoundsBonus_Events_NextEventType = _NextEventType;
	Net_RoundsBonus_Events_NextEventAnnounced = _NextEventAnnounced;
	Net_RoundsBonus_Events_NextEventStartTime = _NextEventStartTime;
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/** Show or hide the UI
 *
 *	@param	_Player										The player to update
 *	@param	_Display									True to show
 *																		False to hide
 */
Void SetVisibility(CPlayer _Player, Boolean _Display) {
	if (_Player == Null) return;
	declare CUIConfig UI <=> UIManager.GetUI(_Player);
	if (UI == Null) return;

	declare netwrite Boolean Net_RoundsBonus_Events_IsVisible for UI;
	Net_RoundsBonus_Events_IsVisible = _Display;
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/** Initialize a player
 *
 *	@param	_Player										The player to initialize
 */
Void InitializePlayer(CPlayer _Player) {
	if (_Player == Null) return;

	SetVisibility(_Player, False);
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Unload the library
Void Unload() {
	UpdateEvent(Data::C_EventType_Null, -1, Data::C_EventType_Null, False, -1);

	foreach (Player in AllPlayers) {
		InitializePlayer(Player);
	}
}

// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ //
/// Load the library
Void Load() {
	Unload();
}
