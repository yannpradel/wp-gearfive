ClubStruct::LibStructuresClub_K_Member[] GetVIPListOnMap(Integer _ClubId, Text _SeasonUid, Text _MapUid) {
	declare K_State Trackmania_ClubStore_State for System = C_State_Null;

	if (Trackmania_ClubStore_State.PinnedClub.Club.Id == _ClubId &&
		Trackmania_ClubStore_State.PinnedClub.Club.VIPListOnMap.existskey(_SeasonUid) &&
		Trackmania_ClubStore_State.PinnedClub.Club.VIPListOnMap[_SeasonUid].existskey(_MapUid)
	) {
		return Trackmania_ClubStore_State.PinnedClub.Club.VIPListOnMap[_SeasonUid][_MapUid];
	}

	foreach (ClubIndex => Club in Trackmania_ClubStore_State.ClubsPager[C_Section_AllClubs].Clubs) {
		if (Club.Id == _ClubId && Club.VIPListOnMap.existskey(_SeasonUid) && Club.VIPListOnMap[_SeasonUid].existskey(_MapUid)) {
			return Trackmania_ClubStore_State.ClubsPager[C_Section_AllClubs].Clubs[ClubIndex].VIPListOnMap[_SeasonUid][_MapUid];
		}
	}

	return [];
}
