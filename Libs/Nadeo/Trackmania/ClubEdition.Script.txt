/// Special router function to push the deprecated or the new ClubEdition page (see /trackmania-next/tmnext/-/issues/5762)
/// Kept to centralize the push to this page in case we need it
Void GoToPage(CMlScript _This, Text _CurrentRoutePath, Text _QueryMessage) {
	Router::SetParentPath(_This, RoutePaths::C_Path_ClubEditionInfo, _CurrentRoutePath);
	Router::Push(
		_This,
		RoutePaths::C_Path_ClubEditionInfo,
		[
			RoutePaths::C_Query_Message => _QueryMessage
		],
		False
	);
}
